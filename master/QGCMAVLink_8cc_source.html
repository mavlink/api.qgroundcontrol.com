484uint32_t QGCMAVLink::highLatencyFailuresToMavSysStatus(mavlink_high_latency2_t& highLatency2)
485{
486 struct failure2Sensor_s {
487 HL_FAILURE_FLAG failureBit;
488 MAV_SYS_STATUS_SENSOR sensorBit;
489 };
490
491 static constexpr const failure2Sensor_s rgFailure2Sensor[] = {
492 { HL_FAILURE_FLAG_GPS, MAV_SYS_STATUS_SENSOR_GPS },
493 { HL_FAILURE_FLAG_DIFFERENTIAL_PRESSURE, MAV_SYS_STATUS_SENSOR_DIFFERENTIAL_PRESSURE },
494 { HL_FAILURE_FLAG_ABSOLUTE_PRESSURE, MAV_SYS_STATUS_SENSOR_ABSOLUTE_PRESSURE },
495 { HL_FAILURE_FLAG_3D_ACCEL, MAV_SYS_STATUS_SENSOR_3D_ACCEL },
496 { HL_FAILURE_FLAG_3D_GYRO, MAV_SYS_STATUS_SENSOR_3D_GYRO },
497 { HL_FAILURE_FLAG_3D_MAG, MAV_SYS_STATUS_SENSOR_3D_MAG },
498 };
499
500 // Map from MAV_FAILURE bits to standard SYS_STATUS message handling
501 uint32_t onboardControlSensorsEnabled = 0;
502 for (size_t i=0; i<sizeof(rgFailure2Sensor)/sizeof(failure2Sensor_s); i++) {
503 const failure2Sensor_s* pFailure2Sensor = &rgFailure2Sensor[i];
504 if (highLatency2.failure_flags & pFailure2Sensor->failureBit) {
505 // Assume if reporting as unhealthy that is it present and enabled
506 onboardControlSensorsEnabled |= pFailure2Sensor->sensorBit;
507 }
508 }
509
510 return onboardControlSensorsEnabled;
511}
