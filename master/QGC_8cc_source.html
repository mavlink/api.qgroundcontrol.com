174QString bigSizeToString(quint64 size)
175{
176 const QLocale locale;
177 if (size < kKB) {
178 return locale.toString(size) + QStringLiteral("B");
179 }
180 if (size < kMB) {
181 return locale.toString(static_cast<double>(size) / kKB, 'f', 1) + QStringLiteral("KB");
182 }
183 if (size < kGB) {
184 return locale.toString(static_cast<double>(size) / kMB, 'f', 1) + QStringLiteral("MB");
185 }
186 if (size < kTB) {
187 return locale.toString(static_cast<double>(size) / kGB, 'f', 1) + QStringLiteral("GB");
188 }
189 return locale.toString(static_cast<double>(size) / kTB, 'f', 1) + QStringLiteral("TB");
190}
