159bool verifySignature(QByteArrayView key, const mavlink_message_t &message)
160{
161 if (key.size() < 32) {
162 return false;
163 }
164
165 // Replicate the C library's signature computation:
166 // SHA-256(secret_key + header_bytes + payload + CRC + link_id + timestamp)
167 // Then compare first 6 bytes with message.signature[7..12]
168
169 const uint8_t* header = reinterpret_cast<const uint8_t*>(&message.magic);
170 const char* payload = reinterpret_cast<const char*>(&message.payload64[0]);
171 const uint8_t* crc = reinterpret_cast<const uint8_t*>(payload + message.len);
172 const uint8_t* sig = message.signature; // [link_id(1), timestamp(6), hash(6)]
173
174 QCryptographicHash sha256(QCryptographicHash::Sha256);
175 sha256.addData(key.first(32));
176 sha256.addData(QByteArrayView(reinterpret_cast<const char*>(header), MAVLINK_NUM_HEADER_BYTES));
177 sha256.addData(QByteArrayView(payload, message.len));
178 sha256.addData(QByteArrayView(reinterpret_cast<const char*>(crc), 2));
179 sha256.addData(QByteArrayView(reinterpret_cast<const char*>(sig), 7)); // link_id + timestamp
180
181 const QByteArray hash = sha256.result();
182 return (memcmp(hash.constData(), sig + 7, 6) == 0);
183}

185QString tryDetectKey(mavlink_channel_t channel, const mavlink_message_t &message)
186{
187 if (!isMessageSigned(message)) {
188 return QString();
189 }
190
191 // If signing is already configured on this channel, the C library already verified it
192 if (isSigningEnabled(channel)) {
193 return QString();
194 }
195
196 auto* signingKeys = MAVLinkSigningKeys::instance();
197 const auto* keys = signingKeys->keys();
198 const int keyCount = keys->count();
199
200 // Try the cached hint for this channel first
201 const QString& hintName = s_channelKeyHint[channel];
202 if (!hintName.isEmpty()) {
203 const QByteArray keyBytes = signingKeys->keyBytesByName(hintName);
204 if (!keyBytes.isEmpty() && verifySignature(keyBytes, message)) {
205 if (initSigning(channel, keyBytes, insecureConnectionAcceptUnsignedCallback)) {
206 qCInfo(MAVLinkSigningLog) << "Auto-detected signing key" << hintName << "on channel" << channel << "(cached hint)";
207 return hintName;
208 }
209 }
210 }
211
212 // Fall back to trying all keys
213 for (int i = 0; i < keyCount; ++i) {
214 const QString name = signingKeys->keyNameAt(i);
215 if (name == hintName) {
216 continue; // already tried above
217 }
218 const QByteArray keyBytes = signingKeys->keyBytesAt(i);
219 if (verifySignature(keyBytes, message)) {
220 if (initSigning(channel, keyBytes, insecureConnectionAcceptUnsignedCallback)) {
221 s_channelKeyHint[channel] = name;
222 qCInfo(MAVLinkSigningLog) << "Auto-detected signing key" << name << "on channel" << channel;
223 return name;
224 }
225 }
226 }
227
228 return QString();
229}
