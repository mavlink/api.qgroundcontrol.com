249int Fact::enumIndex()
250{
251 if (_metaData) {
252 //-- Only enums have an index
253 if (!_metaData->enumValues().isEmpty()) {
254 int index = 0;
255 for (const QVariant &enumValue: _metaData->enumValues()) {
256 if (enumValue == rawValue()) {
257 return index;
258 }
259 //-- Float comparisons don't always work
260 if ((type() == FactMetaData::valueTypeFloat) || (type() == FactMetaData::valueTypeDouble)) {
261 const double diff = fabs(enumValue.toDouble() - rawValue().toDouble());
262 static constexpr double accuracy = 1.0 / 1000000.0;
263 if (diff < accuracy) {
264 return index;
265 }
266 }
267 index++;
268 }
269 // Current value is not in list, add it manually
270 _metaData->addEnumInfo(tr("Unknown: %1").arg(rawValue().toString()), rawValue());
271 emit enumsChanged();

273 }

358QString Fact::_variantToString(const QVariant &variant, int decimalPlaces) const
359{
360 QMutexLocker<QRecursiveMutex> locker(&_rawValueMutex);
361 if (_rawValueIsNotSet) {
362 return invalidValueString(decimalPlaces);
363 }
364 locker.unlock();
365
366 QString valueString;
367
368 const auto stripNegativeZero = [](QString &candidate) {
369 static const QRegularExpression reNegativeZero(QStringLiteral("^-0\\.0+$"));
370 const auto match = reNegativeZero.match(candidate);
371 if (match.hasMatch() || candidate == QStringLiteral("-0")) {
372 candidate = candidate.mid(1);
373 }
374 };
375
376 switch (type()) {
377 case FactMetaData::valueTypeFloat:
378 {
379 const float fValue = variant.toFloat();
380 if (qIsNaN(fValue)) {
381 valueString = invalidValueString(decimalPlaces);
382 } else {
383 valueString = QStringLiteral("%1").arg(fValue, 0, 'f', decimalPlaces);
384 stripNegativeZero(valueString);
385 }
386 }
387 break;
388 case FactMetaData::valueTypeDouble:
389 {
390 const double dValue = variant.toDouble();
391 if (qIsNaN(dValue)) {
392 valueString = invalidValueString(decimalPlaces);
393 } else {
394 valueString = QStringLiteral("%1").arg(dValue, 0, 'f', decimalPlaces);
395 stripNegativeZero(valueString);
396 }
397 break;
398 }
399 case FactMetaData::valueTypeBool:
400 valueString = variant.toBool() ? tr("true") : tr("false");
401 break;
402 case FactMetaData::valueTypeElapsedTimeInSeconds:
403 {
404 const double dValue = variant.toDouble();
405 if (qIsNaN(dValue)) {
406 valueString = invalidValueString(decimalPlaces);
407 } else {
408 QTime time(0, 0, 0, 0);
409 time = time.addSecs(dValue);
410 valueString = time.toString(QStringLiteral("hh:mm:ss"));
411 }
412 break;
413 }
414 default:
415 valueString = variant.toString();
416 break;
417 }
418
419 return valueString;
420}
