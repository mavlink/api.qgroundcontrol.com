16{
17 if (!uri_str) {

19 }
20
21 GstRTSPUrl *url = NULL;

25 return FALSE;
26 }
27
28 res = gst_rtsp_url_parse(uri_str, &url);
29 if ((res != GST_RTSP_OK) || (url == NULL)) {
30 if (url) {
31 gst_rtsp_url_free(url);

33 return FALSE;

37 gst_rtsp_url_free(url);
38 return hasHost;

41bool isHardwareDecoderFactory(GstElementFactory *factory)
42{
43 if (!factory) {
44 return false;
45 }
46
47 const gchar *factoryName = gst_plugin_feature_get_name(GST_PLUGIN_FEATURE(factory));
48 if (!factoryName) {
49 return false;
50 }
51
52 const QString nameLower = QString::fromUtf8(factoryName).toLower();
53
54 // Android MediaCodec: exclude software wrappers, accept remaining as hardware
55 if (nameLower.startsWith("amcviddec-omxgoogle") || nameLower.startsWith("amcviddec-c2android")) {
56 return false;
57 }
58 if (nameLower.startsWith("amcviddec-")) {
59 return true;
60 }
61
62 const auto containsHardware = [](const gchar *value) {
63 if (!value) return false;
64 gchar *lower = g_ascii_strdown(value, -1);
65 bool found = (g_strrstr(lower, "hardware") != nullptr);
66 g_free(lower);
67 return found;
68 };
69
70 if (containsHardware(gst_element_factory_get_metadata(factory, GST_ELEMENT_METADATA_KLASS))) {
71 return true;
72 }
73
74 if (containsHardware(gst_element_factory_get_metadata(factory, GST_ELEMENT_METADATA_DESCRIPTION))) {
75 return true;
76 }
77
78 static constexpr QLatin1String kHardwareTags[] = {
79 QLatin1String("va"),
80 QLatin1String("nv"),
81 QLatin1String("qsv"),
82 QLatin1String("msdk"),
83 QLatin1String("vulkan"),
84 QLatin1String("d3d"),
85 QLatin1String("dxva"),
86 QLatin1String("vtdec"),
87 QLatin1String("metal"),
88 };
89
90 for (const auto &tag : kHardwareTags) {
91 if (nameLower.contains(tag)) {
92 return true;

94 }

118void lowerDecoderRanksByClass(GstRegistry *registry, bool lowerHardware)
119{
120 static constexpr uint16_t NewRank = GST_RANK_NONE;
121 if (!registry) {
122 qCCritical(GStreamerHelpersLog) << "Invalid registry!";
123 return;
124 }
125
126 GList *decoderFactories = gst_element_factory_list_get_elements(
127 static_cast<GstElementFactoryListType>(GST_ELEMENT_FACTORY_TYPE_DECODER | GST_ELEMENT_FACTORY_TYPE_MEDIA_VIDEO),
128 GST_RANK_NONE);
129
130 for (GList *node = decoderFactories; node != nullptr; node = node->next) {

138 }

141 if (!name) {
142 continue;
143 }
144
145 qCDebug(GStreamerHelpersLog) << "Lowering" << (lowerHardware ? "hardware" : "software") << "decoder rank:" << name;

147 }

149 gst_plugin_feature_list_free(decoderFactories);

152void prioritizeByHardwareClass(GstRegistry *registry, uint16_t prioritizedRank, bool requireHardware)
153{
154 if (!registry) {
155 qCCritical(GStreamerHelpersLog) << "Failed to get gstreamer registry.";
156 return;
157 }
158
159 GList *decoderFactories = gst_element_factory_list_get_elements(
160 static_cast<GstElementFactoryListType>(GST_ELEMENT_FACTORY_TYPE_DECODER | GST_ELEMENT_FACTORY_TYPE_MEDIA_VIDEO),
161 GST_RANK_NONE);
162
163 if (!decoderFactories) {
164 qCDebug(GStreamerHelpersLog) << "No decoder factories available while prioritizing"
165 << (requireHardware ? "hardware" : "software") << "decoders";
166 return;
167 }
168
169 qCDebug(GStreamerHelpersLog) << "Prioritizing" << (requireHardware ? "hardware" : "software")
170 << "video decoders with rank:" << prioritizedRank;
171 int matchedFactories = 0;
172 for (GList *node = decoderFactories; node != nullptr; node = node->next) {
173 GstElementFactory *factory = GST_ELEMENT_FACTORY(node->data);
174 if (!factory) {
175 continue;
176 }
177
178 if (GStreamer::isHardwareDecoderFactory(factory) != requireHardware) {
179 continue;
180 }
181
182 const gchar *featureName = gst_plugin_feature_get_name(GST_PLUGIN_FEATURE(factory));
183 if (!featureName) {
184 continue;
185 }
186
187 changeFeatureRank(registry, featureName, prioritizedRank);
188 ++matchedFactories;
189 }
190
191 if (matchedFactories == 0) {
192 qCWarning(GStreamerHelpersLog) << "No" << (requireHardware ? "hardware" : "software")
193 << "video decoder factories found to reprioritize.";
194 }
195
196 qCDebug(GStreamerHelpersLog) << "Lowering" << (requireHardware ? "software" : "hardware") << "decoder ranks.";
197 lowerDecoderRanksByClass(registry, !requireHardware);
198
199 gst_plugin_feature_list_free(decoderFactories);
200}

204void setCodecPriorities(VideoDecoderOptions option)
205{
206 GstRegistry *registry = gst_registry_get();
207
208 if (!registry) {
209 qCCritical(GStreamerHelpersLog) << "Failed to get gstreamer registry.";
210 return;
211 }
212
213 static constexpr uint16_t PrioritizedRank = GST_RANK_PRIMARY + 1;
214
215 switch (option) {
216 case ForceVideoDecoderDefault:
217 // Android/iOS hardware decoders (amcviddec, vtdec) already have higher
218 // default ranks than software decoders, so decodebin tries them first.
219 // However, AMC decoders require GL output (GLMemory caps) which the
220 // current pipeline doesn't support — so they fail and decodebin falls
221 // back to software. Setting software to RANK_NONE here would eliminate
222 // that fallback entirely. Leave ranks untouched until the pipeline
223 // supports GL context sharing with Qt.
224 break;
225 case ForceVideoDecoderSoftware:
226 prioritizeByHardwareClass(registry, PrioritizedRank, false);
227 break;
228 case ForceVideoDecoderHardware:
229 prioritizeByHardwareClass(registry, PrioritizedRank, true);
230 break;
231 case ForceVideoDecoderVAAPI:
232 for (const char *name : {"vaav1dec", "vah264dec", "vah265dec", "vajpegdec", "vampeg2dec", "vavp8dec", "vavp9dec"}) {
233 changeFeatureRank(registry, name, PrioritizedRank);
234 }
235 break;
236 case ForceVideoDecoderNVIDIA:
237 for (const char *name : {"nvav1dec", "nvh264dec", "nvh265dec", "nvjpegdec", "nvmpeg2videodec", "nvmpeg4videodec", "nvmpegvideodec", "nvvp8dec", "nvvp9dec"}) {
238 changeFeatureRank(registry, name, PrioritizedRank);
239 }
240 break;
241 case ForceVideoDecoderDirectX3D:
242 for (const char *name : {"d3d11av1dec", "d3d11h264dec", "d3d11h265dec", "d3d11mpeg2dec", "d3d11vp8dec", "d3d11vp9dec",
243 "d3d12av1dec", "d3d12h264dec", "d3d12h265dec", "d3d12mpeg2dec", "d3d12vp8dec", "d3d12vp9dec",

252 break;
