134bool VideoManager::waitForGStreamerInit(int timeoutMs)
135{
136#ifdef QGC_GST_STREAMING
137 if (_gstreamerDisabledForUnitTests) {
138 return true;
139 }
140
141 if (_initState == InitState::NotStarted) {
142 startGStreamerInit();
143 }
144
145 switch (_initState) {
146 case InitState::Failed:
147 return false;
148 case InitState::GstReady:
149 case InitState::Running:
150 return true;
151 default:
152 break;
153 }
154
155 if (!_gstInitFuture.isValid()) {
156 qCCritical(VideoManagerLog) << "waitForGStreamerInit: no valid future";
157 return false;
158 }
159
160 QEventLoop loop;
161 QTimer timer;
162 timer.setSingleShot(true);
163 QFutureWatcher<bool> watcher;
164 (void) connect(&watcher, &QFutureWatcher<bool>::finished, &loop, &QEventLoop::quit);
165 (void) connect(&timer, &QTimer::timeout, &loop, &QEventLoop::quit);
166
167 watcher.setFuture(_gstInitFuture);
168 if (!watcher.isFinished()) {
169 timer.start(timeoutMs);
170 loop.exec();
171 }
172
173 if (!watcher.isFinished()) {
174 qCCritical(VideoManagerLog) << "Timed out waiting for GStreamer init";
175 return false;
176 }
177
178 const bool success = watcher.result();
179 if (_initState == InitState::Pending || _initState == InitState::QmlReady) {
180 _onGstInitComplete(success);
181 }
182 return _initState != InitState::Failed;
183#else
184 Q_UNUSED(timeoutMs);
185 return true;
186#endif
187}

334void VideoManager::_cleanupOldVideos()
335{
336 if (!SettingsManager::instance()->videoSettings()->enableStorageLimit()->rawValue().toBool()) {
337 return;
338 }
339
340 const QString savePath = SettingsManager::instance()->appSettings()->videoSavePath();
341 QDir videoDir = QDir(savePath);
342 videoDir.setFilter(QDir::Files | QDir::Readable | QDir::NoSymLinks | QDir::Writable);
343 videoDir.setSorting(QDir::Time);
344
345 QStringList nameFilters;
346 for (size_t i = 0; i < std::size(kFileExtension); i++) {
347 nameFilters << QStringLiteral("*.") + kFileExtension[i];
348 }
349
350 videoDir.setNameFilters(nameFilters);
351 QFileInfoList vidList = videoDir.entryInfoList();
352 if (vidList.isEmpty()) {
353 return;
354 }
355
356 uint64_t total = 0;
357 for (const QFileInfo &video : std::as_const(vidList)) {
358 total += video.size();
359 }
360
361 const uint64_t maxSize = SettingsManager::instance()->videoSettings()->maxVideoSize()->rawValue().toUInt() * qPow(1024, 2);
362 while ((total >= maxSize) && !vidList.isEmpty()) {
363 const QFileInfo info = vidList.takeLast();
364 total -= info.size();
365 const QString path = info.filePath();
366 qCDebug(VideoManagerLog) << "Removing old video file:" << path;
367 (void) QFile::remove(path);

642bool VideoManager::_updateAutoStream(VideoReceiver *receiver)
643{
644 const QGCVideoStreamInfo *pInfo = receiver->videoStreamInfo();
645 if (!pInfo) {
646 return false;
647 }
648
649 qCDebug(VideoManagerLog) << QString("Configure stream (%1):").arg(receiver->name()) << pInfo->uri();
650
651 QString source, url;
652 switch (pInfo->type()) {
653 case VIDEO_STREAM_TYPE_RTSP:
654 source = VideoSettings::videoSourceRTSP;
655 url = pInfo->uri();
656 if (source == VideoSettings::videoSourceRTSP) {
657 _videoSettings->rtspUrl()->setRawValue(url);
658 }
659 break;
660 case VIDEO_STREAM_TYPE_TCP_MPEG:
661 source = VideoSettings::videoSourceTCP;
662 url = pInfo->uri();
663 break;
664 case VIDEO_STREAM_TYPE_RTPUDP:
665 if (pInfo->encoding() == VIDEO_STREAM_ENCODING_H265) {
666 source = VideoSettings::videoSourceUDPH265;
667 url = pInfo->uri().contains("udp265://") ? pInfo->uri() : QStringLiteral("udp265://0.0.0.0:%1").arg(pInfo->uri());
668 } else {
669 source = VideoSettings::videoSourceUDPH264;
670 url = pInfo->uri().contains("udp://") ? pInfo->uri() : QStringLiteral("udp://0.0.0.0:%1").arg(pInfo->uri());
671 }
672 break;
673 case VIDEO_STREAM_TYPE_MPEG_TS:
674 source = VideoSettings::videoSourceMPEGTS;
675 url = pInfo->uri().contains("mpegts://") ? pInfo->uri() : QStringLiteral("mpegts://0.0.0.0:%1").arg(pInfo->uri());
676 break;
677 default:
678 qCWarning(VideoManagerLog) << "Unknown VIDEO_STREAM_TYPE";
679 source = VideoSettings::videoSourceNoVideo;
680 url = pInfo->uri();
681 break;
682 }
683
684 const bool settingsChanged = _updateVideoUri(receiver, url);
685 if (settingsChanged) {
686 if (!receiver->isThermal()) {
687 _videoSettings->videoSource()->setRawValue(source);
688 }
689
690 emit autoStreamConfiguredChanged();
691 }
692
693 return settingsChanged;
694}

714bool VideoManager::_updateSettings(VideoReceiver *receiver)
715{
716 if (!receiver) {
717 qCDebug(VideoManagerLog) << "VideoReceiver is NULL";
718 return false;
719 }
720
721 bool settingsChanged = false;
722
723 const bool lowLatency = _videoSettings->lowLatencyMode()->rawValue().toBool();
724 if (lowLatency != receiver->lowLatency()) {
725 receiver->setLowLatency(lowLatency);
726 settingsChanged = true;
727 }
728
729 if (receiver->isThermal()) {
730 return settingsChanged;
731 }
732
733 settingsChanged |= _updateUVC(receiver);
734 settingsChanged |= _updateAutoStream(receiver);
735
736 const QString source = _videoSettings->videoSource()->rawValue().toString();

868void VideoManager::_startReceiver(VideoReceiver *receiver)
869{
870 if (!receiver) {
871 qCDebug(VideoManagerLog) << "VideoReceiver is NULL";
872 return;
873 }
874
875 if (receiver->started()) {
876 qCDebug(VideoManagerLog) << "VideoReceiver is already started" << receiver->name();
877 return;
878 }
879
880 if (receiver->uri().isEmpty()) {
881 qCDebug(VideoManagerLog) << "VideoUri is NULL" << receiver->name();
882 return;
883 }
884
885 const QString source = _videoSettings->videoSource()->rawValue().toString();
886 /* The gstreamer rtsp source will switch to tcp if udp is not available after 5 seconds.
887 So we should allow for some negotiation time for rtsp */
888
889 const uint32_t timeout = ((source == VideoSettings::videoSourceRTSP) ? _videoSettings->rtspTimeout()->rawValue().toUInt() : 3);
890
891 receiver->start(timeout);
