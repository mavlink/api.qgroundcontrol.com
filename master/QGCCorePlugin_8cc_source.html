103void QGCCorePlugin::adjustSettingMetaData(const QString &settingsGroup, FactMetaData &metaData, bool &userVisible)
104{
105#ifdef Q_OS_ANDROID
106 Q_UNUSED(userVisible);
107#endif
108
109 if (settingsGroup == AppSettings::settingsGroup) {
110 if (metaData.name() == AppSettings::indoorPaletteName) {
111 QVariant outdoorPalette;
112#if defined(Q_OS_ANDROID) || defined(Q_OS_IOS)
113 outdoorPalette = 0;
114#else
115 outdoorPalette = 1;
116#endif
117 metaData.setRawDefaultValue(outdoorPalette);
118 return;
119 }
120#if defined(Q_OS_ANDROID) || defined(Q_OS_IOS)
121 else if (metaData.name() == MavlinkSettings::telemetrySaveName) {
122 metaData.setRawDefaultValue(false);
123 return;
124 }
125#endif
126#ifndef Q_OS_ANDROID
127 else if (metaData.name() == AppSettings::androidDontSaveToSDCardName) {
128 userVisible = false;
129 return;
130 }
131#endif
132 }
