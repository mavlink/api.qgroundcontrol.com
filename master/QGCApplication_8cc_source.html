285bool QGCApplication::_initVideo()
286{
287#ifdef QGC_GST_STREAMING
288 // GStreamer video rendering backend selection:
289 // - Windows D3D11: native RHI, no OpenGL needed.
290 // - macOS: appsink → QVideoSink → Metal RHI VideoOutput, no OpenGL needed.
291 // - Linux/other: qml6glsink requires OpenGL. Probe for a working GL context
292 // and fall back to the default graphics API if unavailable.
293 //
294 // The offscreen platform (used in CI boot tests) never provides a real
295 // GL context, so skip the probe there — just set OpenGL API to exercise
296 // the full GStreamer init path.
297 const bool isOffscreen = (qApp->platformName() == QLatin1String("offscreen"));
298
299#if defined(QGC_GST_D3D11_SINK)
300 // D3D11 sink renders via Qt's native D3D11 RHI — no OpenGL needed.
301 if (isOffscreen) {
302 QQuickWindow::setGraphicsApi(QSGRendererInterface::OpenGL);
303 }
304 qCDebug(QGCApplicationLog) << "D3D11 video sink available, using default graphics API";
305#elif defined(Q_OS_MACOS)
306 // macOS Metal rendering path: appsink → QVideoSink → VideoOutput.
307 // Do NOT force OpenGL — let Qt use the default Metal RHI backend.
308 // The appsink path in qgcvideosinkbin avoids the GL-dependent qml6glsink.
309 if (isOffscreen) {
310 QQuickWindow::setGraphicsApi(QSGRendererInterface::OpenGL);
311 } else {
312 qCDebug(QGCApplicationLog) << "macOS: using default RHI backend (Metal) for appsink video path";
313 }
314#else
315 const bool skipGLProbe = isOffscreen;
316
317 if (skipGLProbe) {
318 QQuickWindow::setGraphicsApi(QSGRendererInterface::OpenGL);
319 } else {
320 QOpenGLContext testCtx;
321 if (testCtx.create()) {
322 QQuickWindow::setGraphicsApi(QSGRendererInterface::OpenGL);
323 } else {
324 qCWarning(QGCApplicationLog) << "OpenGL not available; GStreamer video will be disabled."
325 << "Using default graphics API (Metal/Vulkan).";
326 }
327 }
328#endif // QGC_GST_D3D11_SINK / Q_OS_MACOS
329#endif
330
331 QGCCorePlugin::instance(); // CorePlugin must be initialized before VideoManager for Video Cleanup
332 VideoManager *videoManager = VideoManager::instance();
333 videoManager->startGStreamerInit();
334 const bool initSucceeded = !_simpleBootTest || videoManager->waitForGStreamerInit();
335 _videoManagerInitialized = true;
336 return initSucceeded;
337}

548{
549 if (_runningUnitTests) {

551 }

563 download->deleteLater();

568void QGCApplication::_qgcCurrentStableVersionDownloadComplete(bool success, const QString &localFile, const QString &errorMsg)
569{
570 if (success) {
571 QFile versionFile(localFile);
572 if (versionFile.open(QIODevice::ReadOnly)) {
573 QTextStream textStream(&versionFile);
574 const QString version = textStream.readLine();
575
576 qCDebug(QGCApplicationLog) << version;
577
578 int majorVersion, minorVersion, buildVersion;
579 if (_parseVersionText(version, majorVersion, minorVersion, buildVersion)) {
580 if (_majorVersion < majorVersion ||
581 ((_majorVersion == majorVersion) && (_minorVersion < minorVersion)) ||
582 ((_majorVersion == majorVersion) && (_minorVersion == minorVersion) && (_buildVersion < buildVersion))) {
583 showAppMessage(tr("There is a newer version of %1 available. You can download it from %2.").arg(applicationName()).arg(QGCCorePlugin::instance()->stableDownloadLocation()), tr("New Version Available"));
584 }
585 }
586 }
587 } else if (!errorMsg.isEmpty()) {
588 qCDebug(QGCApplicationLog) << "Download QGC stable version failed" << errorMsg;
589 }
590
591 sender()->deleteLater();
592}

594bool QGCApplication::_parseVersionText(const QString &versionString, int &majorVersion, int &minorVersion, int &buildVersion)
595{
596 static const QRegularExpression regExp("v(\\d+)\\.(\\d+)\\.(\\d+)");
597 const QRegularExpressionMatch match = regExp.match(versionString);
598 if (match.hasMatch() && match.lastCapturedIndex() == 3) {
599 majorVersion = match.captured(1).toInt();
600 minorVersion = match.captured(2).toInt();
601 buildVersion = match.captured(3).toInt();
602 return true;
603 }
604
605 return false;
606}

622int QGCApplication::CompressedSignalList::_signalIndex(const QMetaMethod &method)
623{
624 if (method.methodType() != QMetaMethod::Signal) {
625 qCWarning(QGCApplicationLog) << "Internal error:" << Q_FUNC_INFO << "not a signal" << method.methodType();
626 return -1;
627 }
628
629 int index = -1;
630 const QMetaObject *metaObject = method.enclosingMetaObject();
631 for (int i=0; i<=method.methodIndex(); i++) {
632 if (metaObject->method(i).methodType() != QMetaMethod::Signal) {
633 continue;
634 }
635 index++;
636 }
637
638 return index;
639}

681bool QGCApplication::compressEvent(QEvent *event, QObject *receiver, QPostEventList *postedEvents)
682{
683 if (event->type() != QEvent::MetaCall) {
684 return QApplication::compressEvent(event, receiver, postedEvents);
685 }
686
687 const QMetaCallEvent *mce = static_cast<QMetaCallEvent*>(event);
688 if (!mce->sender() || !_compressedSignals.contains(mce->sender()->metaObject(), mce->signalId())) {
689 return QApplication::compressEvent(event, receiver, postedEvents);
690 }
691
692 for (QPostEventList::iterator it = postedEvents->begin(); it != postedEvents->end(); ++it) {
693 QPostEvent &cur = *it;
694 if (cur.receiver != receiver || cur.event == 0 || cur.event->type() != event->type()) {
695 continue;
696 }
697 const QMetaCallEvent *cur_mce = static_cast<QMetaCallEvent*>(cur.event);
698 if (cur_mce->sender() != mce->sender() || cur_mce->signalId() != mce->signalId() || cur_mce->id() != mce->id()) {
699 continue;
700 }
701 /* Keep The Newest Call */
702 // We can't merely qSwap the existing posted event with the new one, since QEvent
703 // keeps track of whether it has been posted. Deletion of a formerly posted event
704 // takes the posted event list mutex and does a useless search of the posted event
705 // list upon deletion. We thus clear the QEvent::posted flag before deletion.
706 struct EventHelper : private QEvent {
707 static void clearPostedFlag(QEvent * ev) {
708 (&static_cast<EventHelper*>(ev)->t)[1] &= ~0x8001; // Hack to clear QEvent::posted
709 }
710 };
711 EventHelper::clearPostedFlag(cur.event);
712 delete cur.event;
713 cur.event = event;
714 return true;
715 }
716
717 return false;
718}

721bool QGCApplication::event(QEvent *e)
722{
723 if (e->type() == QEvent::Quit) {
724 if (!_mainRootWindow) {
725 return QApplication::event(e);
726 }
727 // On OSX if the user selects Quit from the menu (or Command-Q) the ApplicationWindow does not signal closing. Instead you get a Quit event here only.
728 // This in turn causes the standard QGC shutdown sequence to not run. So in this case we close the window ourselves such that the
729 // signal is sent and the normal shutdown sequence runs.
730 const bool forceClose = _mainRootWindow->property("_forceClose").toBool();
731 qCDebug(QGCApplicationLog) << "Quit event" << forceClose;
732 // forceClose
733 // true: Standard QGC shutdown sequence is complete. Let the app quit normally by falling through to the base class processing.
734 // false: QGC shutdown sequence has not been run yet. Don't let this event close the app yet. Close the main window to kick off the normal shutdown.
735 if (!forceClose) {
736 //
737 _mainRootWindow->close();
738 e->ignore();
739 return true;
740 }
741 }
742
743 return QApplication::event(e);
744}

778 } else {
779 qCWarning(QGCApplicationLog) << "Failed to remove test run settings directory:" << settingsAppDir.absolutePath();
