92bool FirmwareImage::_ihxLoad(const QString& ihxFilename)
93{
94 _imageSize = 0;
95 _ihxBlocks.clear();
96
97 QFile ihxFile(ihxFilename);
98 if (!ihxFile.open(QIODevice::ReadOnly | QIODevice::Text)) {
99 emit statusMessage(QString("Unable to open firmware file %1, error: %2").arg(ihxFilename, ihxFile.errorString()));
100 return false;
101 }
102
103 QTextStream stream(&ihxFile);
104
105 while (true) {
106 if (stream.read(1) != ":") {
107 emit statusMessage("Incorrectly formatted .ihx file, line does not begin with :");
108 return false;
109 }
110
111 uint8_t blockByteCount;
112 uint16_t address;
113 uint8_t recordType;
114 QByteArray bytes;
115 uint8_t crc;
116
117 if (!_readByteFromStream(stream, blockByteCount) ||
118 !_readWordFromStream(stream, address) ||
119 !_readByteFromStream(stream, recordType) ||
120 !_readBytesFromStream(stream, blockByteCount, bytes) ||
121 !_readByteFromStream(stream, crc)) {
122 emit statusMessage(tr("Incorrectly formatted line in .ihx file, line too short"));
123 return false;
124 }
125
126 if (!(recordType == 0 || recordType == 1)) {
127 emit statusMessage(tr("Unsupported record type in file: %1").arg(recordType));
128 return false;
129 }
130
131 if (recordType == 0) {
132 bool appendToLastBlock = false;
133
134 // Can we append this block to the last one?
135
136 if (_ihxBlocks.length()) {
137 int lastBlockIndex = _ihxBlocks.length() - 1;
138
139 if (_ihxBlocks[lastBlockIndex].address + _ihxBlocks[lastBlockIndex].bytes.length() == address) {
140 appendToLastBlock = true;
141 }
142 }
143
144 if (appendToLastBlock) {
145 _ihxBlocks[_ihxBlocks.length() - 1].bytes += bytes;
146 // Too noisy even for verbose
147 //qCDebug(FirmwareUpgradeVerboseLog) << QString("_ihxLoad - append - address:%1 size:%2 block:%3").arg(address).arg(blockByteCount).arg(ihxBlockCount());
148 } else {
149 IntelHexBlock_t block;
150
151 block.address = address;
152 block.bytes = bytes;
153
154 _ihxBlocks += block;
155 qCDebug(FirmwareUpgradeVerboseLog) << QString("_ihxLoad - new block - address:%1 size:%2 block:%3").arg(address).arg(blockByteCount).arg(ihxBlockCount());
156 }
157
158 _imageSize += blockByteCount;
159 } else if (recordType == 1) {
160 // EOF
161 qCDebug(FirmwareUpgradeLog) << QString("_ihxLoad - EOF");
162 break;
163 }
164
165 // Move to next line
166 stream.readLine();
167 }
168
169 ihxFile.close();
170
171 return true;
172}

190{
191 _imageSize = 0;
192
193 // We need to collect information from the .px4 file as well as pull the binary image out to a separate file.
194
195 QFile px4File(imageFilename);
196 if (!px4File.open(QIODevice::ReadOnly | QIODevice::Text)) {
197 emit statusMessage(tr("Unable to open firmware file %1, error: %2").arg(imageFilename, px4File.errorString()));
198 return false;
199 }
200
201 QByteArray bytes = px4File.readAll();
202 px4File.close();
203 QJsonDocument doc = QJsonDocument::fromJson(bytes);
204
205 if (doc.isNull()) {
206 emit statusMessage(tr("Supplied file is not a valid JSON document"));
207 return false;
208 }
209
210 QJsonObject px4Json = doc.object();
211
212 // Make sure the keys we need are available
213 QString errorString;
214 QStringList requiredKeys;
215 requiredKeys << _jsonBoardIdKey << _jsonImageKey << _jsonImageSizeKey;
216 if (!JsonParsing::validateRequiredKeys(px4Json, requiredKeys, errorString)) {
217 emit statusMessage(tr("Firmware file missing required key: %1").arg(errorString));
218 return false;
219 }
220
221 // Make sure the keys are the correct type
222 QStringList keys;
223 QList<QJsonValue::Type> types;
224 keys << _jsonBoardIdKey << _jsonParamXmlSizeKey << _jsonParamXmlKey << _jsonAirframeXmlSizeKey << _jsonAirframeXmlKey << _jsonImageSizeKey << _jsonImageKey << _jsonMavAutopilotKey;
225 types << QJsonValue::Double << QJsonValue::Double << QJsonValue::String << QJsonValue::Double << QJsonValue::String << QJsonValue::Double << QJsonValue::String << QJsonValue::Double;
226 if (!JsonParsing::validateKeyTypes(px4Json, keys, types, errorString)) {
227 emit statusMessage(tr("Firmware file has invalid key: %1").arg(errorString));
228 return false;
229 }
230
231 uint32_t firmwareBoardId = (uint32_t)px4Json.value(_jsonBoardIdKey).toInt();
232 if (!isCompatible(_boardId, firmwareBoardId)) {
233 emit statusMessage(tr("Downloaded firmware board id does not match hardware board id: %1 != %2").arg(firmwareBoardId).arg(_boardId));
234 return false;
235 }
236
237 // What firmware type is this?
238 MAV_AUTOPILOT firmwareType = (MAV_AUTOPILOT)px4Json[_jsonMavAutopilotKey].toInt(MAV_AUTOPILOT_PX4);
239 emit statusMessage(QString("MAV_AUTOPILOT = %1").arg(firmwareType));
240
241 // NOTE: PX4 firmware currently embeds parameter metadata as XML
242 // ("parameter_xml" key), but the metadata system now expects JSON.
243 // cacheParameterMetaDataFile() skips non-JSON content (debug-level
244 // log only). This is a no-op until PX4 upstream switches px_mkfw.py
245 // to embed JSON. Modern PX4 vehicles (v1.13+) serve metadata via
246 // the component metadata protocol on connect, so this only affects
247 // pre-v1.13 boards that lack that capability.
248 QByteArray decompressedBytes;
249 bool success = _decompressJsonValue(px4Json, // JSON object
250 bytes, // Raw bytes of JSON document
251 _jsonParamXmlSizeKey, // key which holds byte size
252 _jsonParamXmlKey, // key which holds compressed bytes
253 decompressedBytes); // Returned decompressed bytes

260 if (bytesWritten != decompressedBytes.length()) {
261 emit statusMessage(tr("Write failed for parameter meta data file, error: %1").arg(parameterFile.errorString()));
262 parameterFile.close();
263 QFile::remove(parameterFilename);
264 } else {
265 parameterFile.close();
266 }
267 } else {
268 emit statusMessage(tr("Unable to open parameter meta data file %1 for writing, error: %2").arg(parameterFilename, parameterFile.errorString()));
269 }
270
271 FirmwarePlugin *fwPlugin = FirmwarePluginManager::instance()->firmwarePluginForAutopilot(firmwareType, MAV_TYPE_GENERIC);
272 if (fwPlugin) {
273 fwPlugin->cacheParameterMetaDataFile(parameterFilename);
274 }
275 }
276
277 // Decompress the airframe xml and save to file
278 success = _decompressJsonValue(px4Json, // JSON object
279 bytes, // Raw bytes of JSON document
280 _jsonAirframeXmlSizeKey, // key which holds byte size
281 _jsonAirframeXmlKey, // key which holds compressed bytes
282 decompressedBytes); // Returned decompressed bytes
283 if (success) {

285 //qDebug() << airframeFilename;

289 qint64 bytesWritten = airframeFile.write(decompressedBytes);
290 if (bytesWritten != decompressedBytes.length()) {
291 // FIXME: What about these warnings?
292 emit statusMessage(tr("Write failed for airframe meta data file, error: %1").arg(airframeFile.errorString()));
293 airframeFile.close();
294 QFile::remove(airframeFilename);
295 } else {
296 airframeFile.close();
297 }
298 } else {
299 emit statusMessage(tr("Unable to open airframe meta data file %1 for writing, error: %2").arg(airframeFilename, airframeFile.errorString()));
300 }
301 }
302
303 // Decompress the image and save to file
304 _imageSize = px4Json.value(QString("image_size")).toInt();
305 success = _decompressJsonValue(px4Json, // JSON object
306 bytes, // Raw bytes of JSON document
307 _jsonImageSizeKey, // key which holds byte size
308 _jsonImageKey, // key which holds compressed bytes
309 decompressedBytes); // Returned decompressed bytes

311 return false;
312 }
313

317 }
318

320 QDir imageDir = QFileInfo(imageFilename).dir();

331 emit statusMessage(tr("Write failed for decompressed image file, error: %1").arg(decompressFile.errorString()));
332 return false;
333 }
334 decompressFile.close();
335
336 _binFilename = decompressFilename;
337
338 return true;

342bool FirmwareImage::_decompressJsonValue(const QJsonObject& jsonObject,
343 const QByteArray& jsonDocBytes,
344 const QString& sizeKey,
345 const QString& bytesKey,
346 QByteArray& decompressedBytes)
347{
348 // Validate decompressed size key
349 if (!jsonObject.contains(sizeKey)) {
350 emit statusMessage(QString("Firmware file missing %1 key").arg(sizeKey));
351 return false;
352 }
353 int decompressedSize = jsonObject.value(QString(sizeKey)).toInt();
354 if (decompressedSize == 0) {
355 emit statusMessage(tr("Firmware file has invalid decompressed size for %1").arg(sizeKey));
356 return false;
357 }
358
359 // XXX Qt's JSON string handling is terribly broken, strings
360 // with some length (18K / 25K) are just weirdly cut.
361 // The code below works around this by manually 'parsing'
362 // for the image string. Since its compressed / checksummed
363 // this should be fine.
364
365 QStringList parts = QString(jsonDocBytes).split(QString("\"%1\": \"").arg(bytesKey));
366 if (parts.length() == 1) {
367 emit statusMessage(tr("Could not find compressed bytes for %1 in Firmware file").arg(bytesKey));
368 return false;
369 }
370 parts = parts.last().split("\"");
371 if (parts.length() == 1) {
372 emit statusMessage(tr("Incorrectly formed compressed bytes section for %1 in Firmware file").arg(bytesKey));
373 return false;
374 }
375
376 // Store decompressed size as first four bytes. This is required by qUncompress routine.
377 QByteArray raw;
378 raw.append((unsigned char)((decompressedSize >> 24) & 0xFF));
379 raw.append((unsigned char)((decompressedSize >> 16) & 0xFF));
380 raw.append((unsigned char)((decompressedSize >> 8) & 0xFF));
381 raw.append((unsigned char)((decompressedSize >> 0) & 0xFF));
382
383 QByteArray raw64 = parts.first().toUtf8();
384 raw.append(QByteArray::fromBase64(raw64));
385 decompressedBytes = qUncompress(raw);
386
387 if (decompressedBytes.length() == 0) {
388 emit statusMessage(tr("Firmware file has 0 length %1").arg(bytesKey));
389 return false;
390 }
391 if (decompressedBytes.length() != decompressedSize) {
392 emit statusMessage(tr("Size for decompressed %1 does not match stored size: Expected(%1) Actual(%2)").arg(decompressedSize).arg(decompressedBytes.length()));
393 return false;
394 }
395
396 emit statusMessage(tr("Successfully decompressed %1").arg(bytesKey));
397
398 return true;
399}
