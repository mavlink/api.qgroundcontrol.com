119 "_parameterUpdate" <<

121 "count:" << parameterCount <<
122 "index:" << parameterIndex <<
123 "mavType:" << mavParamType <<
124 "value:" << parameterValue <<

128 // PARAM_REQUEST_LIST. We disregard any of this until the initial request is responded to.
129 if ((parameterIndex == 65535) && (parameterName != QStringLiteral("_HASH_CHECK")) && _paramRequestListTimer.isActive()) {

147 if (_debugCacheMap[componentId].contains(parameterName)) {

174 // This will add the new component id, as well as the the new waiting index and set the retry count for that index to 0
175 _waitingReadParamIndexMap[componentId][waitingIndex] = 0;

216 _updateProgressBar();
217

281{
282 auto paramSetEncoder = [this, componentId, paramName, valueType, rawValue](uint8_t /*systemId*/, uint8_t channel, mavlink_message_t *message) -> void {
283 const MAV_PARAM_TYPE paramType = factTypeToMavType(valueType);
284
285 mavlink_param_union_t union_value{};
286 if (!_fillMavlinkParamUnion(valueType, rawValue, union_value)) {
287 return;
288 }
289
290 char paramId[MAVLINK_MSG_PARAM_SET_FIELD_PARAM_ID_LEN + 1] = {};
291 (void) strncpy(paramId, paramName.toLocal8Bit().constData(), MAVLINK_MSG_PARAM_SET_FIELD_PARAM_ID_LEN);
292
293 (void) mavlink_msg_param_set_pack_chan(
294 MAVLinkProtocol::instance()->getSystemId(),
295 MAVLinkProtocol::getComponentId(),
296 channel,
297 message,
298 static_cast<uint8_t>(_vehicle->id()),
299 static_cast<uint8_t>(componentId),
300 paramId,
301 union_value.param_float,
302 static_cast<uint8_t>(paramType));
303 };
304
305 auto checkForCorrectParamValue = [this, componentId, paramName, rawValue](const mavlink_message_t &message) -> bool {
306 if (message.compid != componentId) {
307 return false;
308 }
309
310 mavlink_param_value_t param_value{};
311 mavlink_msg_param_value_decode(&message, &param_value);
312
313 // This will null terminate the name string
314 char parameterNameWithNull[MAVLINK_MSG_PARAM_VALUE_FIELD_PARAM_ID_LEN + 1] = {};
315 (void) strncpy(parameterNameWithNull, param_value.param_id, MAVLINK_MSG_PARAM_VALUE_FIELD_PARAM_ID_LEN);
316 const QString parameterName(parameterNameWithNull);
317
318 if (parameterName != paramName) {
319 return false;
320 }
321
322 // Check that the value matches what we expect within tolerance, if it doesn't match then this message is not for us
323 QVariant receivedValue;
324 mavlink_param_union_t param_union;
325 param_union.param_float = param_value.param_value;
326 param_union.type = param_value.param_type;
327 if (!_mavlinkParamUnionToVariant(param_union, receivedValue)) {
328 return false;
329 }
330 if (rawValue.typeId() != receivedValue.typeId()) {
331 qCWarning(ParameterManagerLog) << "QVariant type mismatch on PARAM_VALUE ack for" << paramName << ": expected type" << rawValue.typeId() << "got type" << receivedValue.typeId();
332 return false;
333 }
334 if (param_value.param_type == MAV_PARAM_TYPE_REAL32) {
335 // Float comparison must be fuzzy
336 return QGC::fuzzyCompare(rawValue.toFloat(), receivedValue.toFloat());
337 } else {
338 return receivedValue == rawValue;
339 }
340 };
341
342 auto checkForParamError = [componentId, paramName](const mavlink_message_t &message) -> bool {
343 if (message.compid != componentId) {
344 return false;
345 }
346
347 mavlink_param_error_t paramError{};
348 mavlink_msg_param_error_decode(&message, &paramError);
349
350 char paramId[MAVLINK_MSG_PARAM_ERROR_FIELD_PARAM_ID_LEN + 1] = {};
351 (void) strncpy(paramId, paramError.param_id, MAVLINK_MSG_PARAM_ERROR_FIELD_PARAM_ID_LEN);
352
353 return QString(paramId) == paramName;
354 };
355
356 // State Machine:
357 // Send PARAM_SET - 2 retries after initial attempt
358 // Increment pending write count
359 // Wait for PARAM_VALUE ack or PARAM_ERROR rejection
360 // Decrement pending write count
361 //
362 // timeout:
363 // Decrement pending write count
364 // Back up to PARAM_SET for retries
365 //
366 // error (PARAM_ERROR or retries exhausted):
367 // Refresh parameter from vehicle
368 // Notify user of failure
369
370 // Create states
371 auto stateMachine = new QGCStateMachine(QStringLiteral("ParameterManager PARAM_SET"), vehicle(), this);
372 auto sendParamSetState = new SendMavlinkMessageState(stateMachine, paramSetEncoder, kParamSetRetryCount);
373 auto incPendingWriteCountState = new FunctionState(QStringLiteral("ParameterManager increment pending write count"), stateMachine, [this]() {
374 _incrementPendingWriteCount();
375 });
376 auto decPendingWriteCountState = new FunctionState(QStringLiteral("ParameterManager decrement pending write count"), stateMachine, [this]() {
377 _decrementPendingWriteCount();
378 });
379 auto retryDecPendingWriteCountState = new FunctionState(QStringLiteral("ParameterManager retry decrement pending write count"), stateMachine, [this]() {
380 _decrementPendingWriteCount();
381 });
382 auto errorDecPendingWriteCountState = new FunctionState(QStringLiteral("ParameterManager error decrement pending write count"), stateMachine, [this]() {
383 _decrementPendingWriteCount();
384 });
385 auto waitAckState = new WaitForParamResponseState(stateMachine, kWaitForParamValueAckMs, checkForCorrectParamValue, checkForParamError);
386 auto paramRefreshState = new FunctionState(QStringLiteral("ParameterManager param refresh"), stateMachine, [this, componentId, paramName]() {
387 refreshParameter(componentId, paramName);
388 });
389 auto userNotifyState = new FunctionState(QStringLiteral("ParameterManager user notify"), stateMachine, [waitAckState, paramName, this, componentId]() {
390 const QString errorDetail = waitAckState->lastParamErrorString();
391 const QString msg = errorDetail.isEmpty()
392 ? QStringLiteral("Parameter write failed: param: %1 %2").arg(paramName, _vehicleAndComponentString(componentId))
393 : QStringLiteral("Parameter write failed: param: %1 %2 - %3").arg(paramName, _vehicleAndComponentString(componentId), errorDetail);
394 qgcApp()->showAppMessage(msg);
395 });
396 auto logSuccessState = new FunctionState(QStringLiteral("ParameterManager log success"), stateMachine, [this, componentId, paramName]() {
397 qCDebug(ParameterManagerLog) << "Parameter write succeeded: param:" << paramName << _vehicleAndComponentString(componentId);
398 emit _paramSetSuccess(componentId, paramName);
399 });
400 auto logFailureState = new FunctionState(QStringLiteral("ParameterManager log failure"), stateMachine, [this, componentId, paramName]() {
401 qCDebug(ParameterManagerLog) << "Parameter write failed: param:" << paramName << _vehicleAndComponentString(componentId);
402 emit _paramSetFailure(componentId, paramName);
403 });
404 auto finalState = new QGCFinalState(stateMachine);
405
406 // Successful state machine transitions
407 stateMachine->setInitialState(sendParamSetState);
408 sendParamSetState->addThisTransition (&QGCState::advance, incPendingWriteCountState);
409 incPendingWriteCountState->addThisTransition(&QGCState::advance, waitAckState);
410 waitAckState->addThisTransition (&QGCState::advance, decPendingWriteCountState);
411 decPendingWriteCountState->addThisTransition(&QGCState::advance, logSuccessState);
412 logSuccessState->addThisTransition (&QGCState::advance, finalState);
413
414 // Retry transitions (timeout path)
415 waitAckState->addTransition(waitAckState, &WaitStateBase::timeout, retryDecPendingWriteCountState);
416 retryDecPendingWriteCountState->addThisTransition(&QGCState::advance, sendParamSetState);
417
418 // PARAM_ERROR path (definitive rejection - no retries)
419 waitAckState->addThisTransition (&QGCState::error, errorDecPendingWriteCountState);
420 errorDecPendingWriteCountState->addThisTransition (&QGCState::advance, logFailureState);
421
422 // Error transitions (retries exhausted)
423 sendParamSetState->addThisTransition(&QGCState::error, logFailureState);
424
425 // Error state branching transitions
426 logFailureState->addThisTransition (&QGCState::advance, userNotifyState);
427 userNotifyState->addThisTransition (&QGCState::advance, paramRefreshState);
428 paramRefreshState->addThisTransition(&QGCState::advance, finalState);
429
430 qCDebug(ParameterManagerLog) << "Starting state machine for PARAM_SET on: " << paramName << _vehicleAndComponentString(componentId);
431 stateMachine->start();
432}

758bool ParameterManager::_fillIndexBatchQueue(bool waitingParamTimeout)
759{
760 if (!_indexBatchQueueActive) {
761 return false;
762 }
763
764 constexpr int maxBatchSize = 10;
765
766 if (waitingParamTimeout) {
767 // We timed out, clear the queue and try again
768 qCDebug(ParameterManagerLog) << "Refilling index based batch queue due to timeout";
769 _indexBatchQueue.clear();
770 } else {
771 qCDebug(ParameterManagerLog) << "Refilling index based batch queue due to received parameter";
772 }
773
774 for (const int componentId: _waitingReadParamIndexMap.keys()) {
775 if (_waitingReadParamIndexMap[componentId].count()) {
776 qCDebug(ParameterManagerLog) << _logVehiclePrefix(componentId) << "_waitingReadParamIndexMap count" << _waitingReadParamIndexMap[componentId].count();
777 qCDebug(ParameterManagerVerbose1Log) << _logVehiclePrefix(componentId) << "_waitingReadParamIndexMap" << _waitingReadParamIndexMap[componentId];
778 }
779
780 for (const int paramIndex: _waitingReadParamIndexMap[componentId].keys()) {
781 if (_indexBatchQueue.contains(paramIndex)) {
782 // Don't add more than once
783 continue;
784 }
785
786 if (_indexBatchQueue.count() > maxBatchSize) {
787 break;
788 }
789
790 _waitingReadParamIndexMap[componentId][paramIndex]++; // Bump retry count
791 if (_disableAllRetries || (_waitingReadParamIndexMap[componentId][paramIndex] > _maxInitialLoadRetrySingleParam)) {
792 // Give up on this index
793 _failedReadParamIndexMap[componentId] << paramIndex;
794 qCDebug(ParameterManagerLog) << _logVehiclePrefix(componentId) << "Giving up on (paramIndex:" << paramIndex << "retryCount:" << _waitingReadParamIndexMap[componentId][paramIndex] << ")";
795 (void) _waitingReadParamIndexMap[componentId].remove(paramIndex);
796 } else {
797 // Retry again
798 _indexBatchQueue.append(paramIndex);
799 _mavlinkParamRequestRead(componentId, QString(), paramIndex, false /* notifyFailure */);
800 qCDebug(ParameterManagerLog) << _logVehiclePrefix(componentId) << "Read re-request for (paramIndex:" << paramIndex << "retryCount:" << _waitingReadParamIndexMap[componentId][paramIndex] << ")";
801 }
802 }
803 }
804
805 return (!_indexBatchQueue.isEmpty());
806}

808void ParameterManager::_waitingParamTimeout()
809{
810 if (_logReplay) {
811 return;
812 }
813
814 qCDebug(ParameterManagerLog) << _logVehiclePrefix(-1) << "_waitingParamTimeout";
815
816 // Now that we have timed out for possibly the first time we can activate the index batch queue
817 _indexBatchQueueActive = true;
818
819 // First check for any missing parameters from the initial index based load
820 bool paramsRequested = _fillIndexBatchQueue(true /* waitingParamTimeout */);
821 if (!paramsRequested && !_waitingForDefaultComponent && !_mapCompId2FactMap.contains(_vehicle->defaultComponentId())) {
822 // Initial load is complete but we still don't have any default component params. Wait one more cycle to see if the
823 // any show up.
824 qCDebug(ParameterManagerLog) << _logVehiclePrefix(-1) << "Restarting _waitingParamTimeoutTimer - still don't have default component params" << _vehicle->defaultComponentId();
825 _waitingParamTimeoutTimer.start();
826 _waitingForDefaultComponent = true;
827 return;
828 }
829 _waitingForDefaultComponent = false;
830
831 _checkInitialLoadComplete();
832
833 if (paramsRequested) {
834 qCDebug(ParameterManagerLog) << _logVehiclePrefix(-1) << "Restarting _waitingParamTimeoutTimer - re-request";
835 _waitingParamTimeoutTimer.start();
836 }

865void ParameterManager::_mavlinkParamRequestRead(int componentId, const QString &paramName, int paramIndex, bool notifyFailure)
866{
867 auto paramRequestReadEncoder = [this, componentId, paramName, paramIndex](uint8_t /*systemId*/, uint8_t channel, mavlink_message_t *message) -> void {
868 char paramId[MAVLINK_MSG_PARAM_REQUEST_READ_FIELD_PARAM_ID_LEN + 1] = {};
869 (void) strncpy(paramId, paramName.toLocal8Bit().constData(), MAVLINK_MSG_PARAM_REQUEST_READ_FIELD_PARAM_ID_LEN);
870
871 (void) mavlink_msg_param_request_read_pack_chan(MAVLinkProtocol::instance()->getSystemId(), // QGC system id
872 MAVLinkProtocol::getComponentId(), // QGC component id
873 channel,
874 message,
875 static_cast<uint8_t>(_vehicle->id()),
876 static_cast<uint8_t>(componentId),
877 paramId,
878 static_cast<int16_t>(paramIndex));
879 };
880
881 auto checkForCorrectParamValue = [componentId, paramName, paramIndex](const mavlink_message_t &message) -> bool {
882 if (message.compid != componentId) {
883 return false;
884 }
885
886 mavlink_param_value_t param_value{};
887 mavlink_msg_param_value_decode(&message, &param_value);
888
889 // This will null terminate the name string
890 char parameterNameWithNull[MAVLINK_MSG_PARAM_VALUE_FIELD_PARAM_ID_LEN + 1] = {};
891 (void) strncpy(parameterNameWithNull, param_value.param_id, MAVLINK_MSG_PARAM_VALUE_FIELD_PARAM_ID_LEN);
892 const QString parameterName(parameterNameWithNull);
893
894 // Check that this is for the parameter we requested
895 if (paramIndex != -1) {
896 // Index based request
897 if (param_value.param_index != paramIndex) {
898 return false;
899 }
900 } else {
901 // Name based request
902 if (parameterName != paramName) {
903 return false;
904 }
905 }
906
907 return true;
908 };
909
910 auto checkForParamError = [componentId, paramName, paramIndex](const mavlink_message_t &message) -> bool {
911 if (message.compid != componentId) {
912 return false;
913 }
914
915 mavlink_param_error_t paramError{};
916 mavlink_msg_param_error_decode(&message, &paramError);
917
918 char paramId[MAVLINK_MSG_PARAM_ERROR_FIELD_PARAM_ID_LEN + 1] = {};
919 (void) strncpy(paramId, paramError.param_id, MAVLINK_MSG_PARAM_ERROR_FIELD_PARAM_ID_LEN);
920
921 if (paramIndex != -1) {
922 return paramError.param_index == paramIndex;
923 } else {
924 return QString(paramId) == paramName;
925 }
926 };
927
928 // State Machine:
929 // Send PARAM_REQUEST_READ - 2 retries after initial attempt
930 // Wait for PARAM_VALUE ack or PARAM_ERROR rejection
931 //
932 // timeout:
933 // Back up to PARAM_REQUEST_READ for retries
934 //
935 // error (PARAM_ERROR or retries exhausted):
936 // Notify user of failure
937
938 // Create states
939 auto stateMachine = new QGCStateMachine(QStringLiteral("PARAM_REQUEST_READ"), vehicle(), this);
940 auto sendParamRequestReadState = new SendMavlinkMessageState(stateMachine, paramRequestReadEncoder, kParamRequestReadRetryCount);
941 auto waitAckState = new WaitForParamResponseState(stateMachine, kWaitForParamValueAckMs, checkForCorrectParamValue, checkForParamError);
942 auto userNotifyState = new FunctionState(QStringLiteral("User notify"), stateMachine, [waitAckState, paramName, this, componentId]() {
943 const QString errorDetail = waitAckState->lastParamErrorString();
944 const QString msg = errorDetail.isEmpty()
945 ? QStringLiteral("Parameter read failed: param: %1 %2").arg(paramName, _vehicleAndComponentString(componentId))
946 : QStringLiteral("Parameter read failed: param: %1 %2 - %3").arg(paramName, _vehicleAndComponentString(componentId), errorDetail);
947 qgcApp()->showAppMessage(msg);
948 });
949 auto logSuccessState = new FunctionState(QStringLiteral("Log success"), stateMachine, [this, componentId, paramName, paramIndex]() {
950 qCDebug(ParameterManagerLog) << "PARAM_REQUEST_READ succeeded: name:" << paramName << "index" << paramIndex << _vehicleAndComponentString(componentId);
951 emit _paramRequestReadSuccess(componentId, paramName, paramIndex);
952 });
953 auto logFailureState = new FunctionState(QStringLiteral("Log failure"), stateMachine, [this, componentId, paramName, paramIndex]() {
954 qCDebug(ParameterManagerLog) << "PARAM_REQUEST_READ failed: param:" << paramName << "index" << paramIndex << _vehicleAndComponentString(componentId);
955 emit _paramRequestReadFailure(componentId, paramName, paramIndex);
956 });
957 auto finalState = new QGCFinalState(stateMachine);
958
959 // Successful state machine transitions
960 stateMachine->setInitialState(sendParamRequestReadState);
961 sendParamRequestReadState->addThisTransition(&QGCState::advance, waitAckState);
962 waitAckState->addThisTransition (&QGCState::advance, logSuccessState);
963 logSuccessState->addThisTransition (&QGCState::advance, finalState);
964
965 // Retry transitions (timeout path)
966 waitAckState->addTransition(waitAckState, &WaitStateBase::timeout, sendParamRequestReadState);
967
968 // PARAM_ERROR path (definitive rejection - no retries)
969 waitAckState->addThisTransition(&QGCState::error, logFailureState);
970
971 // Error transitions (retries exhausted)
972 sendParamRequestReadState->addThisTransition(&QGCState::error, logFailureState);
973
974 // Error state branching transitions
975 if (notifyFailure) {
976 logFailureState->addThisTransition (&QGCState::advance, userNotifyState);
977 } else {
978 logFailureState->addThisTransition (&QGCState::advance, finalState);
979 }
980 userNotifyState->addThisTransition (&QGCState::advance, finalState);
981
982 qCDebug(ParameterManagerLog) << "Starting state machine for PARAM_REQUEST_READ on: " << paramName << _vehicleAndComponentString(componentId);
983 stateMachine->start();
984}

1018void ParameterManager::_tryCacheHashLoad(int vehicleId, int componentId, const QVariant &hashValue)
1019{
1020 qCDebug(ParameterManagerLog) << "Attemping load from cache";
1021
1022 /* The datastructure of the cache table */
1023 CacheMapName2ParamTypeVal cacheMap;
1024 QFile cacheFile(parameterCacheFile(vehicleId, componentId));
1025 if (!cacheFile.exists()) {
1026 qCDebug(ParameterManagerLog) << "No parameter cache file";
1027 if (!_hashCheckDone) {
1028 _hashCheckDone = true;
1029 if (_cacheOnlyHashCheck) {
1030 qCDebug(ParameterManagerLog) << "Cache-only hash check: no cache file, signalling failure";
1031 emit cacheCheckOnlyFailed();
1032 return;
1033 }
1034 // Standalone hash check path — fall back to PARAM_REQUEST_LIST
1035 _startParameterDownload(MAV_COMP_ID_ALL);
1036 }
1037 // If already in PARAM_REQUEST_LIST flow, just let the stream continue
1038 return;
1039 }
1040 (void) cacheFile.open(QIODevice::ReadOnly);
1041
1042 /* Deserialize the parameter cache table */
1043 QDataStream ds(&cacheFile);
1044 ds >> cacheMap;
1045
1046 /* compute the crc of the local cache to check against the remote */
1047 uint32_t crc32_value = 0;
1048 for (const QString &name: cacheMap.keys()) {
1049 const ParamTypeVal &paramTypeVal = cacheMap[name];
1050 const FactMetaData::ValueType_t factType = static_cast<FactMetaData::ValueType_t>(paramTypeVal.first);
1051
1052 if (_vehicle->compInfoManager()->compInfoParam(MAV_COMP_ID_AUTOPILOT1)->factMetaDataForName(name, factType)->volatileValue()) {
1053 // Does not take part in CRC
1054 qCDebug(ParameterManagerLog) << "Volatile parameter" << name;
1055 } else {
1056 const void *const vdat = paramTypeVal.second.constData();
1057 const FactMetaData::ValueType_t cacheFactType = static_cast<FactMetaData::ValueType_t>(paramTypeVal.first);
1058 crc32_value = QGC::crc32(reinterpret_cast<const uint8_t *>(qPrintable(name)), name.length(), crc32_value);
1059 crc32_value = QGC::crc32(static_cast<const uint8_t *>(vdat), FactMetaData::typeToSize(cacheFactType), crc32_value);
1060 }
1061 }
1062
1063 /* if the two param set hashes match, just load from the disk */
1064 if (crc32_value == hashValue.toUInt()) {
1065 _hashCheckDone = true;
1066 _paramRequestListTimer.stop();
1067 qCDebug(ParameterManagerLog) << "Parameters loaded from cache" << qPrintable(QFileInfo(cacheFile).absoluteFilePath());
1068
1069 const int count = cacheMap.count();
1070 int index = 0;
1071 for (const QString &name: cacheMap.keys()) {
1072 const ParamTypeVal &paramTypeVal = cacheMap[name];
1073 const FactMetaData::ValueType_t factType = static_cast<FactMetaData::ValueType_t>(paramTypeVal.first);
1074 const MAV_PARAM_TYPE mavParamType = factTypeToMavType(factType);
1075 _handleParamValue(componentId, name, count, index++, mavParamType, paramTypeVal.second);
1076 }
1077
1078 const SharedLinkInterfacePtr sharedLink = _vehicle->vehicleLinkManager()->primaryLink().lock();
1079 if (sharedLink) {
1080 mavlink_param_set_t p{};
1081 mavlink_param_union_t union_value{};
1082
1083 // Return the hash value to notify we don't want any more updates
1084 p.param_type = MAV_PARAM_TYPE_UINT32;
1085 (void) strncpy(p.param_id, "_HASH_CHECK", sizeof(p.param_id));
1086 union_value.param_uint32 = crc32_value;
1087 p.param_value = union_value.param_float;
1088 p.target_system = static_cast<uint8_t>(_vehicle->id());
1089 p.target_component = static_cast<uint8_t>(componentId);
1090
1091 mavlink_message_t msg{};
1092 (void) mavlink_msg_param_set_encode_chan(MAVLinkProtocol::instance()->getSystemId(),
1093 MAVLinkProtocol::getComponentId(),
1094 sharedLink->mavlinkChannel(),
1095 &msg,
1096 &p);
1097 (void) _vehicle->sendMessageOnLinkThreadSafe(sharedLink.get(), msg);
1098 }
1099
1100 // Give the user some feedback things loaded properly
1101 QVariantAnimation *const ani = new QVariantAnimation(this);
1102 ani->setEasingCurve(QEasingCurve::OutCubic);
1103 ani->setStartValue(0.0);
1104 ani->setEndValue(1.0);
1105 ani->setDuration(750);
1106
1107 (void) connect(ani, &QVariantAnimation::valueChanged, this, [this](const QVariant &value) {
1108 _setLoadProgress(value.toDouble());
1109 });
1110
1111 // Hide 500ms after animation finishes
1112 connect(ani, &QVariantAnimation::finished, this, [this] {
1113 QTimer::singleShot(500, this, [this] {
1114 _setLoadProgress(0);
1115 });
1116 });
1117
1118 ani->start(QAbstractAnimation::DeleteWhenStopped);
1119 } else {
1120 qCDebug(ParameterManagerLog) << "Parameters cache match failed" << qPrintable(QFileInfo(cacheFile).absoluteFilePath());
1121 if (ParameterManagerDebugCacheFailureLog().isDebugEnabled()) {
1122 _debugCacheCRC[componentId] = true;
1123 _debugCacheMap[componentId] = cacheMap;
1124 for (const QString &name: cacheMap.keys()) {
1125 _debugCacheParamSeen[componentId][name] = false;
1126 }
1127 qgcApp()->showAppMessage(tr("Parameter cache CRC match failed"));
1128 }
1129 if (!_hashCheckDone) {
1130 _hashCheckDone = true;
1131 if (_cacheOnlyHashCheck) {
1132 qCDebug(ParameterManagerLog) << "Cache-only hash check: CRC mismatch, signalling failure";
1133 emit cacheCheckOnlyFailed();
1134 return;
1135 }
1136 // Standalone hash check path — fall back to PARAM_REQUEST_LIST
1137 _startParameterDownload(MAV_COMP_ID_ALL);
1138 }
1139 // If already in PARAM_REQUEST_LIST flow, just let the stream continue
1140 }
1141}

1143QString ParameterManager::readParametersFromStream(QTextStream &stream)
1144{
1145 QString missingErrors;
1146 QString typeErrors;
1147
1148 while (!stream.atEnd()) {
1149 const QString line = stream.readLine();
1150 if (!line.startsWith("#")) {
1151 const QStringList wpParams = line.split("\t");
1152 const int lineMavId = wpParams.at(0).toInt();
1153 if (wpParams.size() == 5) {
1154 if (_vehicle->id() != lineMavId) {
1155 return QStringLiteral("The parameters in the stream have been saved from System Id %1, but the current vehicle has the System Id %2.").arg(lineMavId).arg(_vehicle->id());
1156 }
1157
1158 const int componentId = wpParams.at(1).toInt();
1159 const QString paramName = wpParams.at(2);
1160 const QString valStr = wpParams.at(3);
1161 const uint mavType = wpParams.at(4).toUInt();
1162
1163 if (!parameterExists(componentId, paramName)) {
1164 QString error;
1165 error += QStringLiteral("%1:%2").arg(componentId).arg(paramName);
1166 missingErrors += error;
1167 qCDebug(ParameterManagerLog) << "Skipped due to missing:" << error;
1168 continue;
1169 }
1170
1171 Fact *const fact = getParameter(componentId, paramName);

1396{

1400 }

1402 const int majorVersion = _vehicle->firmwareMajorVersion();
1403 const int minorVersion = _vehicle->firmwareMinorVersion();

1564bool ParameterManager::_parseParamFile(const QString& filename)
1565{
1566 constexpr quint16 magic_standard = 0x671B;
1567 constexpr quint16 magic_withdefaults = 0x671C;
1568 quint32 no_of_parameters_found = 0;
1569 constexpr int componentId = MAV_COMP_ID_AUTOPILOT1;
1570 enum ap_var_type {
1571 AP_PARAM_NONE = 0,
1572 AP_PARAM_INT8,
1573 AP_PARAM_INT16,
1574 AP_PARAM_INT32,
1575 AP_PARAM_FLOAT,
1576 AP_PARAM_VECTOR3F,
1577 AP_PARAM_GROUP
1578 };
1579
1580 qCDebug(ParameterManagerLog) << "_parseParamFile:" << filename;
1581 QFile file(filename);
1582 if (!file.open(QIODevice::ReadOnly)) {
1583 qCDebug(ParameterManagerLog) << "_parseParamFile: Error: Could not open downloaded parameter file.";
1584 return false;
1585 }
1586
1587 QDataStream in(&file);
1588 in.setByteOrder(QDataStream::LittleEndian);
1589
1590 quint16 magic, num_params, total_params;
1591 in >> magic;
1592 in >> num_params;
1593 in >> total_params;
1594
1595 if (in.status() != QDataStream::Ok) {
1596 qCDebug(ParameterManagerLog) << "_parseParamFile: Error: Could not read Header";
1597 goto Error;
1598 }
1599
1600 qCDebug(ParameterManagerVerbose2Log) << "_parseParamFile: magic: 0x" << Qt::hex << magic;
1601 qCDebug(ParameterManagerVerbose2Log) << "_parseParamFile: num_params:" << num_params
1602 << "total_params:" << total_params;
1603
1604 if ((magic != magic_standard) && (magic != magic_withdefaults)) {
1605 qCDebug(ParameterManagerLog) << "_parseParamFile: Error: File does not start with Magic";
1606 goto Error;
1607 }
1608 if (num_params > total_params) {
1609 qCDebug(ParameterManagerLog) << "_parseParamFile: Error: total_params > num_params";
1610 goto Error;
1611 }
1612 if (num_params != total_params) {
1613 /* We requested all parameters, so this is an error here */
1614 qCDebug(ParameterManagerLog) << "_parseParamFile: Error: total_params != num_params";
1615 goto Error;
1616 }
1617
1618 while (in.status() == QDataStream::Ok) {
1619 quint8 byte = 0;
1620 quint8 flags = 0;
1621 quint8 ptype = 0;
1622 quint8 name_len = 0;
1623 quint8 common_len = 0;
1624 bool withdefault = false;
1625 int no_read = 0;
1626 char name_buffer[17];
1627
1628 while (byte == 0x0) { // Eat padding bytes
1629 in >> byte;
1630 if (in.status() != QDataStream::Ok) {
1631 if (no_of_parameters_found == num_params) {
1632 goto Success;
1633 } else {
1634 qCDebug(ParameterManagerLog) << "_parseParamFile: Error: unexpected EOF"
1635 << "number of parameters expected:" << num_params
1636 << "actual:" << no_of_parameters_found;
1637 goto Error;
1638 }
1639 }
1640 }
1641 ptype = byte & 0x0F;
1642 flags = (byte >> 4) & 0x0F;
1643 withdefault = (flags & 0x01) == 0x01;
1644 in >> byte;
1645 if (in.status() != QDataStream::Ok) {
1646 qCritical(ParameterManagerLog) << "_parseParamFile: Error: Unexpected EOF while reading flags";
1647 goto Error;
1648 }
1649 name_len = ((byte >> 4) & 0x0F) + 1;
1650 common_len = byte & 0x0F;
1651 if ((name_len + common_len) > 16) {
1652 qCritical(ParameterManagerLog) << "_parseParamFile: Error: common_len + name_len > 16"
1653 << "name_len" << name_len
1654 << "common_len" << common_len;
1655 goto Error;
1656 }
1657 no_read = in.readRawData(&name_buffer[common_len], static_cast<int>(name_len));
1658 if (no_read != name_len) {
1659 qCritical(ParameterManagerLog) << "_parseParamFile: Error: Unexpected EOF while reading parameterName"
1660 << "Expected:" << name_len
1661 << "Actual:" << no_read;
1662 goto Error;
1663 }
1664 name_buffer[common_len + name_len] = '\0';
1665 const QString parameterName(name_buffer);
1666 qCDebug(ParameterManagerVerbose2Log) << "_parseParamFile: parameter" << parameterName
1667 << "name_len" << name_len
1668 << "common_len" << common_len
1669 << "ptype" << ptype
1670 << "flags" << flags;
1671
1672 QVariant parameterValue = 0;
1673 QVariant defaultValue;
1674 switch (static_cast<ap_var_type>(ptype)) {
1675 qint8 data8;
1676 qint16 data16;
1677 qint32 data32;
1678 float dfloat;
1679 case AP_PARAM_INT8:
1680 in >> data8;
1681 parameterValue = data8;
1682 if (withdefault) {
1683 in >> data8;
1684 defaultValue = data8;
1685 }
1686 break;
1687 case AP_PARAM_INT16:
1688 in >> data16;
1689 parameterValue = data16;
1690 if (withdefault) {
1691 in >> data16;
1692 defaultValue = data16;
1693 }
1694 break;
1695 case AP_PARAM_INT32:
1696 in >> data32;
1697 parameterValue = data32;
1698 if (withdefault) {
1699 in >> data32;
1700 defaultValue = data32;
1701 }
1702 break;
1703 case AP_PARAM_FLOAT:
1704 in >> data32;
1705 (void) memcpy(&dfloat, &data32, 4);
1706 parameterValue = dfloat;
1707 if (withdefault) {
1708 in >> data32;
1709 float ddefault;
1710 (void) memcpy(&ddefault, &data32, 4);
1711 defaultValue = ddefault;
1712 }
1713 break;
1714 default:
1715 qCDebug(ParameterManagerLog) << "_parseParamFile: Error: type is out of range" << ptype;
1716 goto Error;
1717 break;
1718 }
1719 qCDebug(ParameterManagerVerbose2Log) << "paramValue" << parameterValue;
1720
1721 if (++no_of_parameters_found > num_params) {
1722 qCDebug(ParameterManagerLog) << "_parseParamFile: Error: more parameters in file than expected."
1723 << "Expected:" << num_params
1724 << "Actual:" << no_of_parameters_found;
1725 goto Error;
1726 }
1727
1728 const FactMetaData::ValueType_t factType = ((ptype == AP_PARAM_INT8) ? FactMetaData::valueTypeInt8 :
1729 (ptype == AP_PARAM_INT16) ? FactMetaData::valueTypeInt16 :
1730 (ptype == AP_PARAM_INT32) ? FactMetaData::valueTypeInt32 :
1731 FactMetaData::valueTypeFloat);
1732
1733 Fact *fact = nullptr;
1734 if (_mapCompId2FactMap.contains(componentId) && _mapCompId2FactMap[componentId].contains(parameterName)) {
1735 fact = _mapCompId2FactMap[componentId][parameterName];
1736 if (withdefault && defaultValue.isValid()) {
1737 fact->metaData()->setRawDefaultValue(defaultValue);
1738 }
1739 } else {
1740 qCDebug(ParameterManagerVerbose1Log) << _logVehiclePrefix(componentId) << "Adding new fact" << parameterName;
1741
1742 fact = new Fact(componentId, parameterName, factType, this);
1743 FactMetaData *const factMetaData = _vehicle->compInfoManager()->compInfoParam(componentId)->factMetaDataForName(parameterName, fact->type());
1744 fact->setMetaData(factMetaData);
1745
1746 _mapCompId2FactMap[componentId][parameterName] = fact;
1747
1748 // We need to know when the fact value changes so we can update the vehicle
1749 (void) connect(fact, &Fact::containerRawValueChanged, this, &ParameterManager::_factRawValueUpdated);
1750
1751 // Set default before emitting factAdded so QML sees defaultValueAvailable from the start
1752 if (withdefault && defaultValue.isValid()) {
1753 fact->metaData()->setRawDefaultValue(defaultValue);
1754 }
1755
1756 emit factAdded(componentId, fact);
1757 }
1758 fact->containerSetRawValue(parameterValue);
1759 }
1760
1761Success:
1762 file.close();
1763 /* Create empty waiting lists as we have all parameters */
1764 _paramCountMap[componentId] = num_params;
1765 _totalParamCount += num_params;
1766 _waitingReadParamIndexMap[componentId] = QMap<int, int>();
1767 _checkInitialLoadComplete();
1768 _setLoadProgress(0.0);
1769 return true;
1770
1771Error:
1772 file.close();
1773 return false;
1774}
