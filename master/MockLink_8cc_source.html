308 QFile paramFile;
309 if (_firmwareType == MAV_AUTOPILOT_ARDUPILOTMEGA) {
310 if (_vehicleType == MAV_TYPE_FIXED_WING) {
311 paramFile.setFileName(":/FirmwarePlugin/APM/Plane.OfflineEditing.params");

323 const bool success = paramFile.open(QFile::ReadOnly);

325 Q_ASSERT(success);
326
327 QTextStream paramStream(&paramFile);
328 while (!paramStream.atEnd()) {
329 const QString line = paramStream.readLine();
330
331 if (line.startsWith("#")) {

333 }
334
335 const QStringList paramData = line.split("\t");
336 Q_ASSERT(paramData.count() == 5);

338 const int compId = paramData.at(1).toInt();
339 const QString paramName = paramData.at(2);

341 const uint paramType = paramData.at(4).toUInt();
342
343 QVariant paramValue;
344 switch (paramType) {
345 case MAV_PARAM_TYPE_REAL32:
346 paramValue = QVariant(valStr.toFloat());
347 break;
348 case MAV_PARAM_TYPE_UINT32:
349 paramValue = QVariant(valStr.toUInt());
350 break;
351 case MAV_PARAM_TYPE_INT32:
352 paramValue = QVariant(valStr.toInt());
353 break;
354 case MAV_PARAM_TYPE_UINT16:
355 paramValue = QVariant((quint16)valStr.toUInt());
356 break;
357 case MAV_PARAM_TYPE_INT16:
358 paramValue = QVariant((qint16)valStr.toInt());
359 break;
360 case MAV_PARAM_TYPE_UINT8:

363 case MAV_PARAM_TYPE_INT8:

365 break;

370 }

895uint32_t MockLink::_computeParamHash(int componentId) const
896{
897 // Volatile parameters are excluded from the hash, matching PX4 firmware and ParameterManager::_tryCacheHashLoad
898 static const QStringList volatileParams = {
899 QStringLiteral("COM_FLIGHT_UUID"),
900 QStringLiteral("EKF2_MAGBIAS_X"),
901 QStringLiteral("EKF2_MAGBIAS_Y"),
902 QStringLiteral("EKF2_MAGBIAS_Z"),
903 QStringLiteral("EKF2_MAG_DECL"),
904 QStringLiteral("LND_FLIGHT_T_HI"),
905 QStringLiteral("LND_FLIGHT_T_LO"),
906 QStringLiteral("SYS_RESTART_TYPE"),
907 };
908
909 uint32_t crc = 0;
910 const auto &params = _mapParamName2Value[componentId];
911 for (auto it = params.constBegin(); it != params.constEnd(); ++it) {
912 const QString &name = it.key();
913 if (volatileParams.contains(name)) {
914 continue;
915 }
916 const QVariant &value = it.value();
917 const MAV_PARAM_TYPE mavType = _mapParamName2MavParamType[componentId][name];
918 const FactMetaData::ValueType_t factType = ParameterManager::mavTypeToFactType(mavType);
919
920 crc = QGC::crc32(reinterpret_cast<const uint8_t *>(qPrintable(name)), name.length(), crc);
921 crc = QGC::crc32(static_cast<const uint8_t *>(value.constData()), FactMetaData::typeToSize(factType), crc);
922 }
923 return crc;
924}

952void MockLink::_paramRequestListWorker()
953{
954 if (_currentParamRequestListComponentIndex == -1) {
955 // Initial request complete
956 return;
957 }
958
959 // Thread safety: Lock mutex before accessing shared state modified by main thread
960 QMutexLocker locker(&_paramRequestListMutex);
961
962 // Use cached lists instead of calling keys() on every iteration (500Hz)
963 if (_currentParamRequestListComponentIndex >= _paramRequestListComponentIds.count()) {
964 _currentParamRequestListComponentIndex = -1;
965 return;
966 }
967
968 const int componentId = _paramRequestListComponentIds.at(_currentParamRequestListComponentIndex);
969 const int cParameters = _paramRequestListParamNames.count();
970
971 if (_currentParamRequestListParamIndex >= cParameters) {
972 // All regular params sent — for PX4, append _HASH_CHECK as the last entry in the stream.
973 // Uses param_count=0, param_index=-1 (same as standalone response) so ParameterManager
974 // handles it via the _HASH_CHECK early-return path without affecting param count tracking.
975 if (_firmwareType == MAV_AUTOPILOT_PX4 && !_paramRequestListHashCheckSent) {
976 _paramRequestListHashCheckSent = true;
977
978 mavlink_param_union_t valueUnion{};
979 valueUnion.type = MAV_PARAM_TYPE_UINT32;
980 valueUnion.param_uint32 = _computeParamHash(componentId);
981
982 char paramId[MAVLINK_MSG_ID_PARAM_VALUE_LEN]{};
983 (void) strncpy(paramId, "_HASH_CHECK", MAVLINK_MSG_ID_PARAM_VALUE_LEN);
984
985 qCDebug(MockLinkLog) << "Sending _HASH_CHECK in PARAM_REQUEST_LIST stream" << componentId << "hash:" << valueUnion.param_uint32;
986
987 mavlink_message_t responseMsg{};
988 (void) mavlink_msg_param_value_pack_chan(
989 _vehicleSystemId,
990 componentId,
991 mavlinkChannel(),
992 &responseMsg,
993 paramId,
994 valueUnion.param_float,
995 MAV_PARAM_TYPE_UINT32,
996 0, // param_count: 0 to avoid affecting ParameterManager's count tracking
997 -1 // param_index: -1 signals this is a virtual/out-of-band parameter
998 );
999 respondWithMavlinkMessage(responseMsg);
1000 return;
1001 }
1002
1003 // Move to next component
1004 if (++_currentParamRequestListComponentIndex >= _paramRequestListComponentIds.count()) {
1005 _currentParamRequestListComponentIndex = -1;
1006 _paramRequestListComponentIds.clear();
1007 _paramRequestListParamNames.clear();
1008 } else {
1009 // Cache param names for the new component
1010 _paramRequestListParamNames = _mapParamName2Value[_paramRequestListComponentIds.at(_currentParamRequestListComponentIndex)].keys();
1011 _currentParamRequestListParamIndex = 0;
1012 _paramRequestListHashCheckSent = false;
1013 }
1014 return;
1015 }
1016
1017 const QString &paramName = _paramRequestListParamNames.at(_currentParamRequestListParamIndex);
1018
1019 if (((_failureMode == MockConfiguration::FailMissingParamOnInitialRequest) || (_failureMode == MockConfiguration::FailMissingParamOnAllRequests)) && (paramName == _failParam)) {
1020 qCDebug(MockLinkLog) << "Skipping param send:" << paramName;
1021 } else {
1022 char paramId[MAVLINK_MSG_ID_PARAM_VALUE_LEN]{};
1023 mavlink_message_t responseMsg{};
1024
1025 Q_ASSERT(_mapParamName2Value[componentId].contains(paramName));
1026 Q_ASSERT(_mapParamName2MavParamType[componentId].contains(paramName));
1027
1028 const MAV_PARAM_TYPE paramType = _mapParamName2MavParamType[componentId][paramName];
1029
1030 Q_ASSERT(paramName.length() <= MAVLINK_MSG_ID_PARAM_VALUE_LEN);
1031 (void) strncpy(paramId, paramName.toLocal8Bit().constData(), MAVLINK_MSG_ID_PARAM_VALUE_LEN);
1032
1033 qCDebug(MockLinkLog) << "Sending msg_param_value" << componentId << paramId << paramType << _mapParamName2Value[componentId][paramId];
1034
1035 (void) mavlink_msg_param_value_pack_chan(
1036 _vehicleSystemId,
1037 componentId, // component id
1038 mavlinkChannel(),
1039 &responseMsg, // Outgoing message
1040 paramId, // Parameter name
1041 _floatUnionForParam(componentId, paramName), // Parameter value
1042 paramType, // MAV_PARAM_TYPE
1043 cParameters, // Total number of parameters
1044 _currentParamRequestListParamIndex // Index of this parameter
1045 );
1046 respondWithMavlinkMessage(responseMsg);
1047 }
1048
1049 // Move to next param index
1050 ++_currentParamRequestListParamIndex;
1051}

1304 static bool firstCmdUser4 = true;
1305

1307 mavlink_msg_command_long_decode(&msg, &request);
1308
1309 uint8_t commandResult = MAV_RESULT_UNSUPPORTED;
1310
1311 switch (request.command) {
1312 case MAV_CMD_COMPONENT_ARM_DISARM:
1313 if (request.param1 == 0.0f) {
1314 _mavBaseMode &= ~MAV_MODE_FLAG_SAFETY_ARMED;
1315 } else {
1316 _mavBaseMode |= MAV_MODE_FLAG_SAFETY_ARMED;
1317 }
1318 commandResult = MAV_RESULT_ACCEPTED;
1319 break;
1320 case MAV_CMD_PREFLIGHT_CALIBRATION:
1321 _handlePreFlightCalibration(request);
1322 commandResult = MAV_RESULT_ACCEPTED;
1323 break;
1324 case MAV_CMD_DO_MOTOR_TEST:
1325 commandResult = MAV_RESULT_ACCEPTED;
1326 break;
1327 case MAV_CMD_CONTROL_HIGH_LATENCY:
1328 if (linkConfiguration()->isHighLatency()) {
1329 _highLatencyTransmissionEnabled = static_cast<int>(request.param1) != 0;
1330 emit highLatencyTransmissionEnabledChanged(_highLatencyTransmissionEnabled);
1331 commandResult = MAV_RESULT_ACCEPTED;
1332 } else {
1333 commandResult = MAV_RESULT_FAILED;
1334 }
1335 break;
1336 case MAV_CMD_PREFLIGHT_STORAGE:
1337 commandResult = MAV_RESULT_ACCEPTED;
1338 break;
1339 case MAV_CMD_REQUEST_AUTOPILOT_CAPABILITIES:

1356 }
1357 case MAV_CMD_NAV_TAKEOFF:

1359 commandResult = MAV_RESULT_ACCEPTED;
1360 break;
1361 case MAV_CMD_MOCKLINK_ALWAYS_RESULT_ACCEPTED:
1362 // Test command which always returns MAV_RESULT_ACCEPTED
1363 commandResult = MAV_RESULT_ACCEPTED;
1364 break;
1365 case MAV_CMD_MOCKLINK_ALWAYS_RESULT_FAILED:

1369 case MAV_CMD_MOCKLINK_SECOND_ATTEMPT_RESULT_ACCEPTED:

1835void MockLink::_handleLogRequestData(const mavlink_message_t &msg)
1836{
1837 mavlink_log_request_data_t request{};
1838 mavlink_msg_log_request_data_decode(&msg, &request);
1839
1840#ifdef QGC_UNITTEST_BUILD
1841 if (_logDownloadFilename.isEmpty()) {
1842 _logDownloadFilename = _createRandomFile(_logDownloadFileSize);
1843 }
1844#endif
1845
1846 if (request.id != 0) {
1847 qCWarning(MockLinkLog) << "_handleLogRequestData id must be 0";

1849 }
1850

1852 qCWarning(MockLinkLog) << "_handleLogRequestData offset past end of file request.ofs:size" << request.ofs << _logDownloadFileSize;

1855
1856 // This will trigger _logDownloadWorker to send data
1857 // Thread-safe access: Main thread writes, worker thread reads every 2ms. Serialize to avoid

1859 QMutexLocker locker(&_logDownloadMutex);
1860 _logDownloadCurrentOffset = request.ofs;
1861 if (request.ofs + request.count > _logDownloadFileSize) {
1862 request.count = _logDownloadFileSize - request.ofs;
1863 }
1864 _logDownloadBytesRemaining = request.count;

1866
1867void MockLink::_logDownloadWorker()
1868{
1869 // Runs every 2ms (500Hz on worker thread). Must protect shared state modified by main thread.
1870 // Without lock: main could write new offset/count while we're reading, causing corrupted downloads.
1871 QMutexLocker locker(&_logDownloadMutex);
1872 if (_logDownloadBytesRemaining == 0) {
1873 return;
1874 }
1875
1876 QFile file(_logDownloadFilename);
1877 if (!file.open(QIODevice::ReadOnly)) {
1878 qCWarning(MockLinkLog) << "_logDownloadWorker open failed" << file.errorString();
1879 return;
1880 }
1881
1882 uint8_t buffer[MAVLINK_MSG_LOG_DATA_FIELD_DATA_LEN]{};
1883
