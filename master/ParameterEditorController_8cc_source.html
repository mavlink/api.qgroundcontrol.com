218 for (int i=0; i<_categories.count(); i++) {
219 ParameterEditorCategory* category = _categories.value<ParameterEditorCategory*>(i);
220 if (category->name == "Standard" && i != 0) {
221 _categories.removeAt(i);

231 if (i != _categories.count() - 1) {
232 _categories.removeAt(i);

235 break;

381 QTextStream stream(&file);

383 int firstComponentId = -1;
384 while (!stream.atEnd()) {
385 QString line = stream.readLine();
386 if (!line.startsWith("#")) {
387 QStringList wpParams = line.split("\t");

398 bool readOnly = false;
399
400 if (_vehicle->id() != vehicleId) {
401 _diffOtherVehicle = true;
402 }
403 if (firstComponentId == -1) {
404 firstComponentId = componentId;
405 } else if (firstComponentId != componentId) {
406 _diffMultipleComponents = true;
407 }
408
409 if (_parameterMgr->parameterExists(componentId, paramName)) {
410 Fact* vehicleFact = _parameterMgr->getParameter(componentId, paramName);

413

418 fileFact->setRawValue(fileValueStr);
419 vehicleFactMetaData->setVehicleRebootRequired(vehicleRebootRequired);
420 readOnly = vehicleFact->readOnly();
421
422 if (vehicleFact->rawValue() == fileFact->rawValue()) {

424 }
425 fileValueStr = fileFact->enumOrValueString();
426 fileValueVar = fileFact->rawValue();

428 units = vehicleFact->cookedUnits();

432

434 ParameterEditorDiff* paramDiff = new ParameterEditorDiff(this);
435
436 paramDiff->componentId = componentId;

439 paramDiff->fileValue = fileValueStr;

446 }

496void ParameterEditorController::_performSearch(void)
497{
498 QObjectList newParameterList;
499
500 QStringList rgSearchStrings = _searchText.split(' ', Qt::SkipEmptyParts);
501
502 if (rgSearchStrings.isEmpty() && !_showModifiedOnly && !_showFavoritesOnly) {
503 ParameterEditorCategory* category = _categories.count() ? _categories.value<ParameterEditorCategory*>(0) : nullptr;
504 setCurrentCategory(category);
505 _searchParameters.clear();
506 } else {
507 QVector<QRegularExpression> regexList;
508 regexList.reserve(rgSearchStrings.size());
509 for (const QString &searchItem : rgSearchStrings) {
510 QRegularExpression re(searchItem, QRegularExpression::CaseInsensitiveOption);
511 regexList.append(re.isValid() ? re : QRegularExpression());
512 }
513
514 _searchParameters.beginReset();
515 _searchParameters.clear();
516
517 for (int compId : _parameterMgr->componentIds()) {
518 for (const QString &paraName: _parameterMgr->parameterNames(compId)) {
519 Fact* fact = _parameterMgr->getParameter(compId, paraName);
520 bool matched = _shouldShow(fact);
521 // All of the search items must match in order for the parameter to be added to the list
522 if (matched) {
523 for (int i = 0; i < rgSearchStrings.size(); ++i) {
524 const QRegularExpression &re = regexList.at(i);
525 if (re.isValid()) {
526 if (!fact->name().contains(re) &&
527 !fact->shortDescription().contains(re) &&
528 !fact->longDescription().contains(re)) {
529 matched = false;
530 }
531 } else {
532 const QString &searchItem = rgSearchStrings.at(i);
533 if (!fact->name().contains(searchItem, Qt::CaseInsensitive) &&
534 !fact->shortDescription().contains(searchItem, Qt::CaseInsensitive) &&
535 !fact->longDescription().contains(searchItem, Qt::CaseInsensitive)) {
536 matched = false;
537 }
538 }
539 }
540 }
541 if (matched) {
542 _searchParameters.append(fact);
543 }
544 }
545 }
546
547 _searchParameters.endReset();
548
549 if (_parameters != &_searchParameters) {
550 _parameters = &_searchParameters;
551 emit parametersChanged();
552
553 _currentCategory = nullptr;
554 _currentGroup = nullptr;
555 }
556 }
