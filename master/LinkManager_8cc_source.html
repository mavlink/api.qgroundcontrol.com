94bool LinkManager::createConnectedLink(SharedLinkConfigurationPtr &config)
95{
96 SharedLinkInterfacePtr link = nullptr;
97
98 switch(config->type()) {
99#ifndef QGC_NO_SERIAL_LINK
100 case LinkConfiguration::TypeSerial:
101 link = std::make_shared<SerialLink>(config);
102 break;
103#endif
104 case LinkConfiguration::TypeUdp:
105 link = std::make_shared<UDPLink>(config);
106 break;
107 case LinkConfiguration::TypeTcp:
108 link = std::make_shared<TCPLink>(config);
109 break;
110 case LinkConfiguration::TypeBluetooth:
111 link = std::make_shared<BluetoothLink>(config);
112 break;
113 case LinkConfiguration::TypeLogReplay:
114 link = std::make_shared<LogReplayLink>(config);
115 break;
116#ifdef QT_DEBUG
117 case LinkConfiguration::TypeMock:
118 link = std::make_shared<MockLink>(config);
119 break;
120#endif
121 case LinkConfiguration::TypeLast:
122 default:
123 break;
124 }
125
126 if (!link) {
127 return false;
128 }
129
130 if (!link->_allocateMavlinkChannel()) {
131 qCWarning(LinkManagerLog) << "Link failed to setup mavlink channels";
132 return false;
133 }
134
135 // Set up signal connections before adding to list, so link is fully initialized
136 (void) connect(link.get(), &LinkInterface::communicationError, this, &LinkManager::_communicationError);
137 (void) connect(link.get(), &LinkInterface::bytesReceived, MAVLinkProtocol::instance(), &MAVLinkProtocol::receiveBytes);
138 (void) connect(link.get(), &LinkInterface::bytesSent, MAVLinkProtocol::instance(), &MAVLinkProtocol::logSentBytes);
139 (void) connect(link.get(), &LinkInterface::disconnected, this, &LinkManager::_linkDisconnected);
140
141 MAVLinkProtocol::instance()->resetMetadataForLink(link.get());
142
143 // Try to connect before adding to active links list
144 if (!link->_connect()) {
145 (void) disconnect(link.get(), &LinkInterface::communicationError, this, &LinkManager::_communicationError);
146 (void) disconnect(link.get(), &LinkInterface::bytesReceived, MAVLinkProtocol::instance(), &MAVLinkProtocol::receiveBytes);
147 (void) disconnect(link.get(), &LinkInterface::bytesSent, MAVLinkProtocol::instance(), &MAVLinkProtocol::logSentBytes);
148 (void) disconnect(link.get(), &LinkInterface::disconnected, this, &LinkManager::_linkDisconnected);
149 link->_freeMavlinkChannel();
150 config->setLink(nullptr);
151 return false;
152 }
153
154 {
155 QMutexLocker locker(&_linksMutex);
156 _rgLinks.append(link);
157 }
158 config->setLink(link);
159
160 return true;
161}

306void LinkManager::loadLinkConfigurationList()
307{
308 QSettings settings;
309 // Is the group even there?
310 if (settings.contains(LinkConfiguration::settingsRoot() + "/count")) {
311 // Find out how many configurations we have
312 const int count = settings.value(LinkConfiguration::settingsRoot() + "/count").toInt();
313 for (int i = 0; i < count; i++) {
314 const QString root = LinkConfiguration::settingsRoot() + QStringLiteral("/Link%1").arg(i);
315 if (!settings.contains(root + "/type")) {
316 qCWarning(LinkManagerLog) << "Link Configuration" << root << "has no type.";
317 continue;
318 }
319
320 LinkConfiguration::LinkType type = static_cast<LinkConfiguration::LinkType>(settings.value(root + "/type").toInt());
321 if (type >= LinkConfiguration::TypeLast) {
322 qCWarning(LinkManagerLog) << "Link Configuration" << root << "an invalid type:" << type;
323 continue;
324 }
325
326 if (!settings.contains(root + "/name")) {
327 qCWarning(LinkManagerLog) << "Link Configuration" << root << "has no name.";
328 continue;
329 }
330
331 const QString name = settings.value(root + "/name").toString();
332 if (name.isEmpty()) {
333 qCWarning(LinkManagerLog) << "Link Configuration" << root << "has an empty name.";
334 continue;
335 }
336
337 LinkConfiguration* link = nullptr;
338 switch(type) {
339#ifndef QGC_NO_SERIAL_LINK
340 case LinkConfiguration::TypeSerial:
341 link = new SerialConfiguration(name);
342 break;
343#endif
344 case LinkConfiguration::TypeUdp:
345 link = new UDPConfiguration(name);
346 break;
347 case LinkConfiguration::TypeTcp:
348 link = new TCPConfiguration(name);
349 break;
350 case LinkConfiguration::TypeBluetooth:
351 link = new BluetoothConfiguration(name);
352 break;
353 case LinkConfiguration::TypeLogReplay:
354 link = new LogReplayConfiguration(name);
355 break;
356#ifdef QT_DEBUG
357 case LinkConfiguration::TypeMock:
358 link = new MockConfiguration(name);
359 break;
360#endif
361 case LinkConfiguration::TypeLast:
362 default:
363 break;
364 }
365
366 if (link) {
367 const bool autoConnect = settings.value(root + "/auto").toBool();
368 link->setAutoConnect(autoConnect);
369 const bool highLatency = settings.value(root + "/high_latency").toBool();
370 link->setHighLatency(highLatency);
371 link->loadSettings(settings, root);
372 addConfiguration(link);
373 }
374 }
375 }
376
377 // Enable automatic Serial PX4/3DR Radio hunting
378 _configurationsLoaded = true;

622uint8_t LinkManager::allocateMavlinkChannel()
623{
624 for (uint8_t mavlinkChannel = 0; mavlinkChannel < MAVLINK_COMM_NUM_BUFFERS; mavlinkChannel++) {
625 if (_mavlinkChannelsUsedBitMask & (1 << mavlinkChannel)) {
626 continue;
627 }
628
629 mavlink_reset_channel_status(mavlinkChannel);
630 mavlink_status_t* const mavlinkStatus = mavlink_get_channel_status(mavlinkChannel);
631 mavlinkStatus->flags |= MAVLINK_STATUS_FLAG_OUT_MAVLINK1;
632 _mavlinkChannelsUsedBitMask |= (1 << mavlinkChannel);
633 qCDebug(LinkManagerLog) << "allocateMavlinkChannel" << mavlinkChannel;
634 return mavlinkChannel;
635 }
636
637 qWarning(LinkManagerLog) << "allocateMavlinkChannel: all channels reserved!";
638 return invalidMavlinkChannel();
639}

704void LinkManager::_filterCompositePorts(QList<QGCSerialPortInfo> &portList)
705{
706 typedef QPair<quint16, quint16> VidPidPair_t;
707
708 QMap<VidPidPair_t, QStringList> seenSerialNumbers;
709
710 for (auto it = portList.begin(); it != portList.end();) {
711 const QGCSerialPortInfo &portInfo = *it;
712 if (portInfo.hasVendorIdentifier() && portInfo.hasProductIdentifier() && !portInfo.serialNumber().isEmpty() && portInfo.serialNumber() != "0") {
713 VidPidPair_t vidPid(portInfo.vendorIdentifier(), portInfo.productIdentifier());
714 if (seenSerialNumbers.contains(vidPid) && seenSerialNumbers[vidPid].contains(portInfo.serialNumber())) {
715 // Some boards are a composite USB device, with the first port being mavlink and the second something else. We only expose to first mavlink port.
716 // However internal NMEA devices can present like this, so dont skip anything with NMEA in description
717 if(!portInfo.description().contains("NMEA")) {
718 qCDebug(LinkManagerVerboseLog) << QStringLiteral("Removing secondary port on same device - port:%1 vid:%2 pid%3 sn:%4").arg(portInfo.portName()).arg(portInfo.vendorIdentifier()).arg(portInfo.productIdentifier()).arg(portInfo.serialNumber());
719 it = portList.erase(it);
720 continue;
721 }
722 }
723 seenSerialNumbers[vidPid].append(portInfo.serialNumber());
724 }
725 it++;
726 }
727}

729void LinkManager::_addSerialAutoConnectLink()
730{
731 QList<QGCSerialPortInfo> portList;
732#ifdef Q_OS_ANDROID
733 // Android builds only support a single serial connection. Repeatedly calling availablePorts after that one serial
734 // port is connected leaks file handles due to a bug somewhere in android serial code. In order to work around that
735 // bug after we connect the first serial port we stop probing for additional ports.
736 if (!_isSerialPortConnected()) {
737 portList = QGCSerialPortInfo::availablePorts();
738 }
739#else
740 portList = QGCSerialPortInfo::availablePorts();
741#endif
742
743 _filterCompositePorts(portList);
744
745 QStringList currentPorts;
746 for (const QGCSerialPortInfo &portInfo: portList) {
747 qCDebug(LinkManagerVerboseLog) << "-----------------------------------------------------";
748 qCDebug(LinkManagerVerboseLog) << "portName: " << portInfo.portName();
749 qCDebug(LinkManagerVerboseLog) << "systemLocation: " << portInfo.systemLocation();
750 qCDebug(LinkManagerVerboseLog) << "description: " << portInfo.description();
751 qCDebug(LinkManagerVerboseLog) << "manufacturer: " << portInfo.manufacturer();
752 qCDebug(LinkManagerVerboseLog) << "serialNumber: " << portInfo.serialNumber();
753 qCDebug(LinkManagerVerboseLog) << "vendorIdentifier: " << portInfo.vendorIdentifier();
754 qCDebug(LinkManagerVerboseLog) << "productIdentifier: " << portInfo.productIdentifier();
755
756 currentPorts << portInfo.systemLocation();
757
758 QGCSerialPortInfo::BoardType_t boardType;
759 QString boardName;
760
761 // check to see if nmea gps is configured for current Serial port, if so, set it up to connect
762 if (portInfo.systemLocation().trimmed() == _autoConnectSettings->autoConnectNmeaPort()->cookedValueString()) {
763 if (portInfo.systemLocation().trimmed() != _nmeaDeviceName) {
764 _nmeaDeviceName = portInfo.systemLocation().trimmed();
765 qCDebug(LinkManagerLog) << "Configuring nmea port" << _nmeaDeviceName;
766 QSerialPort* newPort = new QSerialPort(portInfo, this);
767 _nmeaBaud = _autoConnectSettings->autoConnectNmeaBaud()->cookedValue().toUInt();
768 newPort->setBaudRate(static_cast<qint32>(_nmeaBaud));
769 qCDebug(LinkManagerLog) << "Configuring nmea baudrate" << _nmeaBaud;
770 // This will stop polling old device if previously set
771 QGCPositionManager::instance()->setNmeaSourceDevice(newPort);
772 if (_nmeaPort) {
773 delete _nmeaPort;
774 }
775 _nmeaPort = newPort;
776 } else if (_autoConnectSettings->autoConnectNmeaBaud()->cookedValue().toUInt() != _nmeaBaud) {
777 _nmeaBaud = _autoConnectSettings->autoConnectNmeaBaud()->cookedValue().toUInt();
778 _nmeaPort->setBaudRate(static_cast<qint32>(_nmeaBaud));
779 qCDebug(LinkManagerLog) << "Configuring nmea baudrate" << _nmeaBaud;
780 }
781 } else if (portInfo.getBoardInfo(boardType, boardName)) {
782 // Should we be auto-connecting to this board type?
783 if (!_allowAutoConnectToBoard(boardType)) {
784 continue;
785 }
786
787 if (portInfo.isBootloader()) {
788 // Don't connect to bootloader
789 qCDebug(LinkManagerLog) << "Waiting for bootloader to finish" << portInfo.systemLocation();
790 continue;
791 }
792 if (_portAlreadyConnected(portInfo.systemLocation()) || (_autoConnectRTKPort == portInfo.systemLocation())) {
793 qCDebug(LinkManagerVerboseLog) << "Skipping existing autoconnect" << portInfo.systemLocation();
794 } else if (!_autoconnectPortWaitList.contains(portInfo.systemLocation())) {
795 // We don't connect to the port the first time we see it. The ability to correctly detect whether we
796 // are in the bootloader is flaky from a cross-platform standpoint. So by putting it on a wait list
797 // and only connect on the second pass we leave enough time for the board to boot up.
798 qCDebug(LinkManagerLog) << "Waiting for next autoconnect pass" << portInfo.systemLocation() << boardName;
799 _autoconnectPortWaitList[portInfo.systemLocation()] = 1;
800 } else if ((++_autoconnectPortWaitList[portInfo.systemLocation()] * _autoconnectUpdateTimerMSecs) > _autoconnectConnectDelayMSecs) {
801 SerialConfiguration* pSerialConfig = nullptr;
802 _autoconnectPortWaitList.remove(portInfo.systemLocation());
803 switch (boardType) {
804 case QGCSerialPortInfo::BoardTypePixhawk:
805 pSerialConfig = new SerialConfiguration(tr("%1 on %2 (AutoConnect)").arg(boardName, portInfo.portName().trimmed()));
806 pSerialConfig->setUsbDirect(true);
807 break;
808 case QGCSerialPortInfo::BoardTypeSiKRadio:
809 pSerialConfig = new SerialConfiguration(tr("%1 on %2 (AutoConnect)").arg(boardName, portInfo.portName().trimmed()));
810 break;
811 case QGCSerialPortInfo::BoardTypeOpenPilot:
812 pSerialConfig = new SerialConfiguration(tr("%1 on %2 (AutoConnect)").arg(boardName, portInfo.portName().trimmed()));
813 break;
814 case QGCSerialPortInfo::BoardTypeRTKGPS:
815 qCDebug(LinkManagerLog) << "RTK GPS auto-connected" << portInfo.portName().trimmed();
816 _autoConnectRTKPort = portInfo.systemLocation();
817 GPSManager::instance()->gpsRtk()->connectGPS(portInfo.systemLocation(), boardName);
818 break;
819 default:
820 qCWarning(LinkManagerLog) << "Internal error: Unknown board type" << boardType;
821 continue;
822 }
823
824 if (pSerialConfig) {
825 qCDebug(LinkManagerLog) << "New auto-connect port added: " << pSerialConfig->name() << portInfo.systemLocation();
826 pSerialConfig->setBaud((boardType == QGCSerialPortInfo::BoardTypeSiKRadio) ? 57600 : 115200);
827 pSerialConfig->setDynamic(true);
828 pSerialConfig->setPortName(portInfo.systemLocation());
829 pSerialConfig->setAutoConnect(true);
830
831 SharedLinkConfigurationPtr sharedConfig(pSerialConfig);
832 createConnectedLink(sharedConfig);
833 }
834 }
835 }
836 }
837
838 // Check for RTK GPS connection gone
839 if (!_autoConnectRTKPort.isEmpty() && !currentPorts.contains(_autoConnectRTKPort)) {
840 qCDebug(LinkManagerLog) << "RTK GPS disconnected" << _autoConnectRTKPort;
841 GPSManager::instance()->gpsRtk()->disconnectGPS();
842 _autoConnectRTKPort.clear();
843 }
844}

846bool LinkManager::_allowAutoConnectToBoard(QGCSerialPortInfo::BoardType_t boardType) const
847{
848 switch (boardType) {
849 case QGCSerialPortInfo::BoardTypePixhawk:
850 if (_autoConnectSettings->autoConnectPixhawk()->rawValue().toBool()) {
851 return true;
852 }
853 break;
854 case QGCSerialPortInfo::BoardTypeSiKRadio:
855 if (_autoConnectSettings->autoConnectSiKRadio()->rawValue().toBool()) {
856 return true;
857 }
858 break;
859 case QGCSerialPortInfo::BoardTypeOpenPilot:
860 if (_autoConnectSettings->autoConnectLibrePilot()->rawValue().toBool()) {
861 return true;
862 }
863 break;
864 case QGCSerialPortInfo::BoardTypeRTKGPS:
865 if (_autoConnectSettings->autoConnectRTKGPS()->rawValue().toBool() && !GPSManager::instance()->gpsRtk()->connected()) {
866 return true;
867 }
868 break;
869 default:
870 qCWarning(LinkManagerLog) << "Internal error: Unknown board type" << boardType;
871 return false;
872 }
873
874 return false;
875}
