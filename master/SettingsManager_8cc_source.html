110void SettingsManager::_loadSettingsFiles()
111{
112 // Settings files can be found in the settingsSavePath() directory
113 // Settings files are json files which end in the settingsFileExtension extension
114 // The format for a settings file is:
115 // {
116 // "version": 1,
117 // "fileType": "Settings",
118 // "groups": {
119 // "groupName": {
120 // "settingName": {
121 // "forceRawValue": <value>, // Forces the rawValue for this setting to the specific value
122 // any FactMetaData json keys except for name and type,
123 // ...
124 // },
125 // "groupName": {
126 // ...
127 // }
128 // }
129 // }
130
131 QDir settingsDir(_appSettings->settingsSavePath());
132 if (!settingsDir.exists()) {
133 qCWarning(SettingsManagerLog) << "Settings directory does not exist:" << settingsDir.absolutePath();
134 return;
135 }
136
137 QStringList settingsFiles = settingsDir.entryList(QStringList() << QString("*.%1").arg(_appSettings->settingsFileExtension), QDir::Files);
138 for (const QString &fileName : settingsFiles) {
139 QFileInfo fileInfo(settingsDir, fileName);
140 if (!fileInfo.isFile()) continue;
141
142 // Load the settings file
143 qCDebug(SettingsManagerLog) << "Loading settings file:" << fileInfo.absoluteFilePath();
144
145 QJsonDocument jsonDoc;
146 QString errorString;
147 if (!JsonParsing::isJsonFile(fileInfo.absoluteFilePath(), jsonDoc, errorString)) {
148 qCWarning(SettingsManagerLog) << "Failed to load settings file:" << fileInfo.absoluteFilePath() << errorString;
149 continue;
150 }
151
152 QJsonObject jsonObject = jsonDoc.object();
153
154 // Validate the settings file
155 int version;
156 if (!JsonHelper::validateInternalQGCJsonFile(jsonObject, "Settings", 1, 1, version, errorString)) {
157 qCWarning(SettingsManagerLog) << "Settings file failed validation:" << fileInfo.absoluteFilePath() << errorString;
158 continue;
159 }
160
161 // Validate the remainder of the file
162
163 // groups key is an object
164 static const QList<JsonHelper::KeyValidateInfo> keyInfoList = {
165 { kJsonGroupsObjectKey, QJsonValue::Object, true },
166 };
167 if (!JsonHelper::validateKeys(jsonObject, keyInfoList, errorString)) {
168 qCWarning(SettingsManagerLog) << "Settings file incorrect format:" << fileInfo.absoluteFilePath() << errorString;
169 continue;
170 }
171
172 auto groupsObject = jsonObject[kJsonGroupsObjectKey].toObject();
173 for (const QString &groupName : groupsObject.keys()) {
174 qCDebug(SettingsManagerLog) << " Loading settings group:" << groupName;
175
176 const QJsonValue &groupValue = groupsObject[groupName];
177 if (!groupValue.isObject()) {
178 qCWarning(SettingsManagerLog) << "Settings file incorrect format, group is not an object:" << fileInfo.absoluteFilePath()
179 << groupName;
180 continue;
181 }
182
183 auto groupObject = groupValue.toObject();
184 for (const QString &settingName : groupObject.keys()) {
185 qCDebug(SettingsManagerLog) << " Loading settings:" << groupName << settingName;
186
187 if (!groupObject[settingName].isObject()) {
188 qCWarning(SettingsManagerLog) << "Settings file incorrect format, setting is not an object:" << fileInfo.absoluteFilePath()
189 << groupName << settingName;
190 continue;
191 }
192
193 // Store the setting overrides. Note that last one wins if there are multiple settings files with the same setting.
194 QJsonObject metaDataObject = groupObject[settingName].toObject();
195 _settingsFileOverrides[groupName][settingName] = metaDataObject;
196 }
197 }
198 }
199}

201void SettingsManager::adjustSettingMetaData(const QString &settingsGroup, FactMetaData &metaData, bool &userVisible)
202{
203 userVisible = true; // By default all settings are visible
204
205 SettingsManager *settingsManager = SettingsManager::instance();
206 if (!settingsManager) {
207 qCWarning(SettingsManagerLog) << "SettingsManager instance not available";
208 return;
209 }
210
211 if (!QGC::runningUnitTests()) {
212 // Apply settings file overrides
213 const auto &groupOverrides = settingsManager->_settingsFileOverrides;
214 if (groupOverrides.contains(settingsGroup) && groupOverrides[settingsGroup].contains(metaData.name())) {
215 QJsonObject settingOverrideJsonObject = groupOverrides[settingsGroup][metaData.name()];
216
217 // We need to stuff in name and type so settingOverrideJsonObject can parse properly
218 settingOverrideJsonObject["name"] = metaData.name();
219 settingOverrideJsonObject["type"] = FactMetaData::typeToString(metaData.type());
220
221 qCDebug(SettingsManagerLog) << "Applying settings file override for" << settingsGroup << metaData.name();
222
223 QScopedPointer<FactMetaData> overrideMetaData(FactMetaData::createFromJsonObject(settingOverrideJsonObject, {}, nullptr));
224
225 // Apply overrides
226 for (const QString &metaDataName : settingOverrideJsonObject.keys()) {
227 if (metaDataName == kJsonVisibleKey) {
228 qCDebug(SettingsManagerLog) << " Setting visibility to" << settingOverrideJsonObject[kJsonVisibleKey].toBool();
229 userVisible = settingOverrideJsonObject[kJsonVisibleKey].toBool();
230 } else if (metaDataName == kJsonForceRawValueKey) {
231 qCDebug(SettingsManagerLog) << " Setting forceRawValue to" << settingOverrideJsonObject[kJsonForceRawValueKey];
232 metaData.setRawDefaultValue(settingOverrideJsonObject[kJsonForceRawValueKey].toVariant());
233 userVisible = false;
234 } else if (metaDataName == FactMetaData::_defaultValueJsonKey) {
235 qCDebug(SettingsManagerLog) << " Setting default to" << overrideMetaData->rawDefaultValue();
236 metaData.setRawDefaultValue(overrideMetaData->rawDefaultValue());
237 } else if (metaDataName == FactMetaData::_minJsonKey) {
238 qCDebug(SettingsManagerLog) << " Setting min to" << overrideMetaData->rawMin();
239 metaData.setRawMin(overrideMetaData->rawMin());
240 } else if (metaDataName == FactMetaData::_maxJsonKey) {
241 qCDebug(SettingsManagerLog) << " Setting max to" << overrideMetaData->rawMax();
242 metaData.setRawMax(overrideMetaData->rawMax());
243 } else if (metaDataName == FactMetaData::_decimalPlacesJsonKey) {
244 qCDebug(SettingsManagerLog) << " Setting decimalPlaces to" << overrideMetaData->decimalPlaces();
245 metaData.setDecimalPlaces(overrideMetaData->decimalPlaces());
246 } else if (metaDataName == FactMetaData::_enumValuesJsonKey) {
247 qCDebug(SettingsManagerLog) << " Setting enumInfo to" << overrideMetaData->enumValues() << overrideMetaData->enumStrings();
248 metaData.setEnumInfo(overrideMetaData->enumStrings(), overrideMetaData->enumValues());
249 } else if (metaDataName == FactMetaData::_enumBitmaskArrayJsonKey) {
250 qCDebug(SettingsManagerLog) << " Setting bitmaskInfo to" << overrideMetaData->bitmaskValues() << overrideMetaData->bitmaskStrings();
251 metaData.setBitmaskInfo(overrideMetaData->bitmaskStrings(), overrideMetaData->bitmaskValues());
252 } else if (metaDataName == FactMetaData::_longDescriptionJsonKey) {
253 qCDebug(SettingsManagerLog) << " Setting longDesc to" << overrideMetaData->longDescription();
254 metaData.setLongDescription(overrideMetaData->longDescription());
255 } else if (metaDataName == FactMetaData::_shortDescriptionJsonKey) {
256 qCDebug(SettingsManagerLog) << " Setting shortDesc to" << overrideMetaData->shortDescription();
257 metaData.setShortDescription(overrideMetaData->shortDescription());
258 }
259 }
260 }
261 }
262
263 // Give QGCCorePlugin a whack at it too
264 QGCCorePlugin::instance()->adjustSettingMetaData(settingsGroup, metaData, userVisible);
