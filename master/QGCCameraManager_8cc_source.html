142void QGCCameraManager::_mavlinkMessageReceived(const mavlink_message_t &message)
143{
144 if (!_initialConnectComplete) {
145 return;
146 }
147
148 // Only pay attention to camera components (MAV_COMP_ID_CAMERA..CAMERA6)
149 // and camera-related messages proxied by the autopilot.
150 const bool fromAutopilot = message.compid == MAV_COMP_ID_AUTOPILOT1;
151 const bool fromCamera = (message.compid >= MAV_COMP_ID_CAMERA) && (message.compid <= MAV_COMP_ID_CAMERA6);
152 if ((message.sysid == _vehicle->id()) && (fromAutopilot || fromCamera)) {
153 switch (message.msgid) {
154 case MAVLINK_MSG_ID_CAMERA_CAPTURE_STATUS:
155 _handleCameraCaptureStatus(message);
156 break;
157 case MAVLINK_MSG_ID_STORAGE_INFORMATION:
158 _handleStorageInformation(message);
159 break;
160 case MAVLINK_MSG_ID_HEARTBEAT:
161 // Autopilot heartbeats should not be treated as camera discovery.
162 // Only actual camera component heartbeats should start CAMERA_INFORMATION requests.
163 if (fromCamera) {
164 _handleHeartbeat(message);
165 }
166 break;
167 case MAVLINK_MSG_ID_CAMERA_INFORMATION:
168 _handleCameraInfo(message);
169 break;
170 case MAVLINK_MSG_ID_CAMERA_SETTINGS:
171 _handleCameraSettings(message);
172 break;
173 case MAVLINK_MSG_ID_PARAM_EXT_ACK:
174 _handleParamExtAck(message);
175 break;
176 case MAVLINK_MSG_ID_PARAM_EXT_VALUE:
177 _handleParamExtValue(message);
178 break;
179 case MAVLINK_MSG_ID_VIDEO_STREAM_INFORMATION:
180 _handleVideoStreamInformation(message);
181 break;
182 case MAVLINK_MSG_ID_VIDEO_STREAM_STATUS:
183 _handleVideoStreamStatus(message);
184 break;
185 case MAVLINK_MSG_ID_BATTERY_STATUS:
186 _handleBatteryStatus(message);
187 break;
188 case MAVLINK_MSG_ID_CAMERA_TRACKING_IMAGE_STATUS:
189 _handleTrackingImageStatus(message);
190 break;
191 case MAVLINK_MSG_ID_CAMERA_FOV_STATUS:
192 _handleCameraFovStatus(message);
193 break;
194 default:
195 break;
196 }
197 }

200void QGCCameraManager::_handleHeartbeat(const mavlink_message_t &message)
201{
202 const QString sCompID = QString::number(message.compid);
203
204 if (!_cameraInfoRequest.contains(sCompID)) {
205 qCDebug(CameraManagerLog) << "Heartbeat from" << QGCMAVLink::compIdToString(message.compid);
206 CameraStruct *pInfo = new CameraStruct(this, message.compid, _vehicle);
207 pInfo->lastHeartbeat.start();
208 _cameraInfoRequest[sCompID] = pInfo;
209 _requestCameraInfo(pInfo);
210 return;
211 }
212
213 CameraStruct *pInfo = _cameraInfoRequest[sCompID];
214 if (!pInfo) {
215 qCWarning(CameraManagerLog) << sCompID << "is null";
216 return;
217 }
218
219 if (pInfo->infoReceived) {
220 pInfo->lastHeartbeat.start();
221 return;
222 }
223
224 if (pInfo->lastHeartbeat.elapsed() > kSilentTimeoutMs) {
225 qCDebug(CameraManagerLog) << "Camera" << QGCMAVLink::compIdToString(message.compid) << "reappeared after being silent. Resetting retry count and requesting info.";
226 pInfo->retryCount = 0;
227 pInfo->backoffTimer.stop();
228 pInfo->lastHeartbeat.start();
229 _requestCameraInfo(pInfo);
230 return;
231 }
232
233 pInfo->lastHeartbeat.start();
234}

308void QGCCameraManager::_handleCameraInfo(const mavlink_message_t& message)
309{
310 const QString sCompID = QString::number(message.compid);
311 if (!_cameraInfoRequest.contains(sCompID)) {
312 qCDebug(CameraManagerLog) << "Ignoring - Camera info not requested for component" << QGCMAVLink::compIdToString(message.compid);
313 return;
314 }
315 if (_cameraInfoRequest[sCompID]->infoReceived) {
316 qCDebug(CameraManagerLog) << "Ignoring - Already received camera info for component" << QGCMAVLink::compIdToString(message.compid);
317 return;
318 }
319
320 mavlink_camera_information_t info{};
321 mavlink_msg_camera_information_decode(&message, &info);
322 qCDebug(CameraManagerLog) << "Camera information received from" << QGCMAVLink::compIdToString(message.compid)
323 << "Model:" << reinterpret_cast<const char*>(info.model_name);
324 qCDebug(CameraManagerLog) << "Creating MavlinkCameraControlInterface for camera";
325
326 MavlinkCameraControlInterface *pCamera = _vehicle->firmwarePlugin()->createCameraControl(&info, _vehicle, message.compid, this);
327 if (pCamera) {
328 _addCameraControlToLists(pCamera);
329
330 _cameraInfoRequest[sCompID]->infoReceived = true;
331 _cameraInfoRequest[sCompID]->retryCount = 0;
332 _cameraInfoRequest[sCompID]->backoffTimer.stop();
333 qCDebug(CameraManagerLog) << "Success for compId" << QGCMAVLink::compIdToString(message.compid) << "- reset retry counter";
334 }
335
336 double aspect = std::numeric_limits<double>::quiet_NaN();
337
338 if (info.resolution_h > 0 && info.resolution_v > 0) {
339 aspect = double(info.resolution_v) / double(info.resolution_h);
340 } else if (info.sensor_size_h > 0.f && info.sensor_size_v > 0.f) {
341 aspect = double(info.sensor_size_v) / double(info.sensor_size_h);
342 }
343
344 _aspectByCompId.insert(message.compid, aspect);
345}

347void QGCCameraManager::_checkForLostCameras()
348{
349 QList<QString> stale;
350 for (auto it = _cameraInfoRequest.cbegin(), end = _cameraInfoRequest.cend(); it != end; ++it) {
351 const auto *info = it.value();
352 if (info && info->infoReceived && (info->lastHeartbeat.elapsed() > kSilentTimeoutMs)) {
353 stale.push_back(it.key());
354 }
355 }
356 if (stale.isEmpty()) {
357 return;
358 }
359
360 bool removedAny = false;
361 for (const QString& key : std::as_const(stale)) {
362 CameraStruct* pInfo = _cameraInfoRequest.take(key);
363 if (!pInfo) {
364 continue;
365 }
366
367 MavlinkCameraControlInterface* pCamera = _findCamera(pInfo->compID);
368 if (pCamera) {
369 const int idx = _cameras.indexOf(pCamera);
370 if (idx >= 0) {
371 qCDebug(CameraManagerLog) << "Removing lost camera" << QGCMAVLink::compIdToString(pInfo->compID);
372 removedAny = true;
373 (void) _cameraLabels.removeAt(idx);
374 (void) _cameras.removeAt(idx);
375 pCamera->deleteLater();
376 }
377 }
378
379 delete pInfo;
380 }
381
382 if (!removedAny) {
383 return;
384 }
385
386 if (_cameras.isEmpty()) {
387 _addCameraControlToLists(_simulatedCameraControl);
388 }
389
390 emit cameraLabelsChanged();
391 emit camerasChanged();
392
393 if (_currentCameraIndex != 0) {
394 _currentCameraIndex = 0;
395 emit currentCameraChanged();
396 }
397 emit streamChanged();
398}

533static void _requestCameraInfoHelper(QGCCameraManager *manager, QGCCameraManager::CameraStruct *pInfo)
534{
535 // Give up after max attempts
536 if (pInfo->retryCount >= kMaxRetryCount) {
537 qCDebug(CameraManagerLog) << "Giving up requesting camera info after" << pInfo->retryCount << "attempts for compId" << QGCMAVLink::compIdToString(pInfo->compID);
538 return;
539 }
540
541 // Alternate between REQUEST_MESSAGE and REQUEST_CAMERA_INFORMATION
542 if ((pInfo->retryCount % 2) == 0) {
543 qCDebug(CameraManagerLog) << "Using MAV_CMD_REQUEST_MESSAGE:CAMERA_INFORMATION for compId" << QGCMAVLink::compIdToString(pInfo->compID);
544 manager->vehicle()->requestMessage(_requestCameraInfoMessageResultHandler, pInfo, pInfo->compID, MAVLINK_MSG_ID_CAMERA_INFORMATION);
545 } else {
546 qCDebug(CameraManagerLog) << "Using MAV_CMD_REQUEST_CAMERA_INFORMATION for compId" << QGCMAVLink::compIdToString(pInfo->compID);
547
548 Vehicle::MavCmdAckHandlerInfo_t ackHandlerInfo{};
549 ackHandlerInfo.resultHandler = _requestCameraInfoCommandResultHandler;
550 ackHandlerInfo.resultHandlerData = pInfo;
551 ackHandlerInfo.progressHandler = nullptr;
552 ackHandlerInfo.progressHandlerData = nullptr;
553
554 pInfo->vehicle->sendMavCommandWithHandler(&ackHandlerInfo, pInfo->compID, MAV_CMD_REQUEST_CAMERA_INFORMATION, 1 /* request camera capabilities */);
555 }

558static void _handleCameraInfoRetry(QGCCameraManager::CameraStruct *cameraInfo)
559{
560 if (!cameraInfo) {
561 return;
562 }
563
564 QGCCameraManager *manager = cameraInfo->manager;
565 if (!manager) {
566 qCDebug(CameraManagerLog) << "manager is unavailable for compId" << QGCMAVLink::compIdToString(cameraInfo->compID);
567 return;
568 }
569
570 cameraInfo->retryCount++;
571
572 // For even attempts >= 2, use exponential backoff
573 if ((cameraInfo->retryCount >= 2) && ((cameraInfo->retryCount % 2) == 0)) {
574 const int delaySeconds = 1 << (cameraInfo->retryCount / 2);
575 const int delayMs = delaySeconds * 1000;
576
577 qCDebug(CameraManagerLog) << "Waiting" << delaySeconds << "seconds before retry for compId" << QGCMAVLink::compIdToString(cameraInfo->compID);
578
579 cameraInfo->backoffTimer.stop();
580 (void) QObject::disconnect(&cameraInfo->backoffTimer, nullptr, nullptr, nullptr);
581
582 // Capture compID by value and look up the struct
583 const uint8_t compId = cameraInfo->compID;
584 QPointer<QGCCameraManager> mgrGuard(manager);
585
586 (void) QObject::connect(&cameraInfo->backoffTimer, &QTimer::timeout,
587 &cameraInfo->backoffTimer, // context ensures timer is still alive
588 [mgrGuard, compId]() {
589 if (!mgrGuard) {
590 return;
591 }
592
593 auto* info = mgrGuard->findCameraStruct(compId);
594 if (info) {
595 _requestCameraInfoHelper(mgrGuard.data(), info);
596 }
597 });
598
599 cameraInfo->backoffTimer.start(delayMs);
600 } else {
601 _requestCameraInfoHelper(manager, cameraInfo);
602 }
603}

774void QGCCameraManager::_handleCameraFovStatus(const mavlink_message_t& message)
775{
776 mavlink_camera_fov_status_t fov{};
777 mavlink_msg_camera_fov_status_decode(&message, &fov);
778
779 if (!std::isfinite(fov.hfov) || fov.hfov <= 0.0 || fov.hfov >= 180.0) {
780 return;
781 }
782
783 double aspect = aspectForComp(message.compid);
784 if (!std::isfinite(aspect) || aspect <= 0.0) {
785 aspect = 16.0 / 9.0;
786 }
787
788 const double hfovRad = fov.hfov * kPi / 180.0;
789 const double vfovRad = 2.0 * std::atan(std::tan(hfovRad * 0.5) * aspect);
790 const double vfovDeg = vfovRad * 180.0 / kPi;
791
792 if (!std::isfinite(vfovDeg) || vfovDeg <= 0.0 || vfovDeg >= 180.0) {
793 qCWarning(CameraManagerLog) << "Invalid calculated VFOV:" << vfovDeg
794 << "hfov:" << fov.hfov
795 << "aspect:" << aspect
796 << "compId:" << message.compid;
797 return;
798 }
799
800 auto* settings = SettingsManager::instance()->gimbalControllerSettings();
801 settings->cameraHFov()->setRawValue(fov.hfov);
802 settings->cameraVFov()->setRawValue(vfovDeg);
803}
