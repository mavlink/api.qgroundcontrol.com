120void ActuatorTest::ackHandler(MAV_RESULT commandResult, VehicleTypes::MavCmdResultFailureCode_t failureCode)
121{
122 // upon receiving an (n)ack, continuously cycle through the active actuators, one at a time
123 _commandInProgress = false;
124 if (failureCode == VehicleTypes::MavCmdResultFailureNoResponseToCommand) {
125 // on timeout, just try the next one
126 sendNext();
127 } else if (commandResult == MAV_RESULT_ACCEPTED) {
128 if (_currentState != -1 && _states[_currentState].state == ActuatorState::State::Stopping) {
129 _states[_currentState].state = ActuatorState::State::NotActive;
130 }
131 sendNext();
132 } else { // failure
133 if (_currentState != -1) {
134 _states[_currentState].state = ActuatorState::State::NotActive;
135 }
136
137 if (!_hadFailure) {
138 QString message;
139 if (commandResult == MAV_RESULT_TEMPORARILY_REJECTED) {
140 message = tr("Actuator test command temporarily rejected");
141 } else if (commandResult == MAV_RESULT_DENIED) {
142 message = tr("Actuator test command denied");
143 } else if (commandResult == MAV_RESULT_UNSUPPORTED) {
144 message = tr("Actuator test command not supported");
145 } else {
146 message = tr("Actuator test command failed");
