80{

84
85 QGeoCoordinate terrainRequestCoordSWCorner(static_cast<double>(_currentTerrainRequest.lat) / 1e7, static_cast<double>(_currentTerrainRequest.lon) / 1e7);
86 const int spacingBetweenGrids = _currentTerrainRequest.grid_spacing * 4;
87
88 // Each TERRAIN_DATA sent to vehicle contains a 4x4 grid of heights
89 // TERRAIN_REQUEST.mask has a bit for each entry in an 8x7 grid
90 // gridBit = 0 refers to the the sw corner of the 8x7 grid
91
92 bool bitFound = false;
93 for (int rowIndex=0; rowIndex<7; rowIndex++) {
94 for (int colIndex=0; colIndex<8; colIndex++) {
95 const uint8_t gridBit = (rowIndex * 8) + colIndex;
96 const uint64_t checkBit = 1ull << gridBit;
97 if (_currentTerrainRequest.mask & checkBit) {
98 // Move east and then north to generate the coordinate for sw corner of the specific gridBit
99 QGeoCoordinate swCorner = terrainRequestCoordSWCorner.atDistanceAndAzimuth(spacingBetweenGrids * colIndex, 90);
100 swCorner = swCorner.atDistanceAndAzimuth(spacingBetweenGrids * rowIndex, 0);
101 _sendTerrainData(swCorner, gridBit);
102 bitFound = true;
103 break;
104 }
105 }
106
107 if (bitFound) {
108 break;
109 }
110 }
111
112 if (bitFound) {
113 // Kick timer to send next possible TERRAIN_DATA to vehicle
114 _terrainDataSendTimer->start();
115 } else {
116 _terrainRequestActive = false;
117 }
