518void Vehicle::_mavlinkMessageReceived(LinkInterface* link, mavlink_message_t message)
519{
520 if (message.sysid != _systemID && message.sysid != 0) {
521 // We allow RADIO_STATUS messages which come from a link the vehicle is using to pass through and be handled
522 if (!(message.msgid == MAVLINK_MSG_ID_RADIO_STATUS && _vehicleLinkManager->containsLink(link))) {
523 return;
524 }
525 }
526
527 // Try to auto-detect signing key from incoming signed packets
528 if (MAVLinkSigning::isMessageSigned(message) && !MAVLinkSigning::isSigningEnabled(static_cast<mavlink_channel_t>(link->mavlinkChannel()))) {
529 const QString detectedKeyName = MAVLinkSigning::tryDetectKey(static_cast<mavlink_channel_t>(link->mavlinkChannel()), message);
530 if (!detectedKeyName.isEmpty() && link == vehicleLinkManager()->primaryLink().lock().get()) {
531 _mavlinkSigningKeyName = detectedKeyName;
532 emit mavlinkSigningChanged();
533 }
534 }
535
536 // We give the link manager first whack since it it reponsible for adding new links
537 _vehicleLinkManager->mavlinkMessageReceived(link, message);
538
539 //-- Check link status
540 _messagesReceived++;
541 emit messagesReceivedChanged();
542 if(!_heardFrom) {
543 if(message.msgid == MAVLINK_MSG_ID_HEARTBEAT) {
544 _heardFrom = true;
545 _compID = message.compid;
546 _messageSeq = message.seq + 1;
547 }
548 } else {
549 if(_compID == message.compid) {
550 uint16_t seq_received = static_cast<uint16_t>(message.seq);
551 uint16_t packet_lost_count = 0;
552 //-- Account for overflow during packet loss
553 if(seq_received < _messageSeq) {
554 packet_lost_count = (seq_received + 255) - _messageSeq;
555 } else {
556 packet_lost_count = seq_received - _messageSeq;
557 }
558 _messageSeq = message.seq + 1;
559 _messagesLost += packet_lost_count;
560 if(packet_lost_count)
561 emit messagesLostChanged();
562 }
563 }
564
565 // Give the plugin a change to adjust the message contents
566 if (!_firmwarePlugin->adjustIncomingMavlinkMessage(this, &message)) {
567 return;
568 }
569
570 // Give the Core Plugin access to all mavlink traffic
571 if (!QGCCorePlugin::instance()->mavlinkMessage(this, link, message)) {
572 return;
573 }
574
575 if (!_terrainProtocolHandler->mavlinkMessageReceived(message)) {
576 return;
577 }
578 _ftpManager->_mavlinkMessageReceived(message);
579 _parameterManager->mavlinkMessageReceived(message);
580 (void) QMetaObject::invokeMethod(_imageProtocolManager, "mavlinkMessageReceived", Qt::AutoConnection, message);
581 _remoteIDManager->mavlinkMessageReceived(message);
582
583 _reqMsgCoord->handleReceivedMessage(message);
584
585 // Handle creation of dynamic fact group lists
586 _batteryFactGroupListModel->handleMessageForFactGroupCreation(this, message);
587 _escStatusFactGroupListModel->handleMessageForFactGroupCreation(this, message);
588
589 // Let the fact groups take a whack at the mavlink traffic
590 for (FactGroup* factGroup : factGroups()) {
591 factGroup->handleMessage(this, message);
592 }
593
594 this->handleMessage(this, message);
595
596 switch (message.msgid) {
597 case MAVLINK_MSG_ID_HOME_POSITION:
598 _handleHomePosition(message);
599 break;
600 case MAVLINK_MSG_ID_HEARTBEAT:
601 _handleHeartbeat(message);
602 break;
603 case MAVLINK_MSG_ID_RC_CHANNELS:
604 _handleRCChannels(message);
605 break;
606 case MAVLINK_MSG_ID_SERVO_OUTPUT_RAW:
607 {
608 mavlink_servo_output_raw_t servoOutputRaw;
609 mavlink_msg_servo_output_raw_decode(&message, &servoOutputRaw);

611 // ArduPilot commonly publishes servo1_raw..servo16_raw in a single packet (port may remain 0).
612 const uint16_t rawValues[16] = {
613 servoOutputRaw.servo1_raw,
614 servoOutputRaw.servo2_raw,
615 servoOutputRaw.servo3_raw,
616 servoOutputRaw.servo4_raw,
617 servoOutputRaw.servo5_raw,
618 servoOutputRaw.servo6_raw,
619 servoOutputRaw.servo7_raw,
620 servoOutputRaw.servo8_raw,
621 servoOutputRaw.servo9_raw,
622 servoOutputRaw.servo10_raw,
623 servoOutputRaw.servo11_raw,
624 servoOutputRaw.servo12_raw,
625 servoOutputRaw.servo13_raw,
626 servoOutputRaw.servo14_raw,
627 servoOutputRaw.servo15_raw,
628 servoOutputRaw.servo16_raw
629 };
630
631 for (int servoIndex = 0; servoIndex < _servoOutputRawValues.size() && servoIndex < 16; servoIndex++) {
632 _servoOutputRawValues[servoIndex] = (rawValues[servoIndex] == UINT16_MAX) ? -1 : static_cast<int>(rawValues[servoIndex]);
633 }
634
635 emit servoOutputsChanged(_servoOutputRawValues);
636 }
637 break;
638 case MAVLINK_MSG_ID_BATTERY_STATUS:
639 _handleBatteryStatus(message);
640 break;
641 case MAVLINK_MSG_ID_SYS_STATUS:
642 _handleSysStatus(message);
643 break;
644 case MAVLINK_MSG_ID_EXTENDED_SYS_STATE:
645 _handleExtendedSysState(message);
646 break;
647 case MAVLINK_MSG_ID_COMMAND_ACK:
648 _handleCommandAck(message);
649 break;
650 case MAVLINK_MSG_ID_LOGGING_DATA:
651 _handleMavlinkLoggingData(message);

653 case MAVLINK_MSG_ID_LOGGING_DATA_ACKED:

660 _handleGlobalPositionInt(message);

662 case MAVLINK_MSG_ID_CAMERA_IMAGE_CAPTURED:
663 _handleCameraImageCaptured(message);
664 break;
665 case MAVLINK_MSG_ID_ADSB_VEHICLE:
666 ADSBVehicleManager::instance()->mavlinkMessageReceived(message);
667 break;
668 case MAVLINK_MSG_ID_HIGH_LATENCY:
669 _handleHighLatency(message);
670 break;
671 case MAVLINK_MSG_ID_HIGH_LATENCY2:

674 case MAVLINK_MSG_ID_STATUSTEXT:
675 m_statusTextHandler->mavlinkMessageReceived(message);
676 break;
677 case MAVLINK_MSG_ID_ORBIT_EXECUTION_STATUS:
678 _handleOrbitExecutionStatus(message);
679 break;
680 case MAVLINK_MSG_ID_PING:
681 _handlePing(link, message);
682 break;
683 case MAVLINK_MSG_ID_OBSTACLE_DISTANCE:
684 _handleObstacleDistance(message);
685 break;
686 case MAVLINK_MSG_ID_FENCE_STATUS:
687 _handleFenceStatus(message);
688 break;

691 case MAVLINK_MSG_ID_CURRENT_EVENT_SEQUENCE:
692 case MAVLINK_MSG_ID_RESPONSE_EVENT_ERROR:
693 _handleEventMessage(message);
694 break;

723#if !defined(QGC_NO_ARDUPILOT_DIALECT)

752 _handleControlStatus(message);

755 _handleCommandLong(message);
756 break;
757 }

759 // This must be emitted after the vehicle processes the message. This way the vehicle state is up to date when anyone else
760 // does processing.
761 emit mavlinkMessageReceived(message);

1065void Vehicle::_handleSysStatus(mavlink_message_t& message)
1066{
1067 if (message.compid != _defaultComponentId) {
1068 return;
1069 }
1070
1071 mavlink_sys_status_t sysStatus;
1072 mavlink_msg_sys_status_decode(&message, &sysStatus);
1073
1074 _sysStatusSensorInfo->update(sysStatus);
1075
1076 if (sysStatus.onboard_control_sensors_enabled & MAV_SYS_STATUS_PREARM_CHECK) {
1077 if (!_readyToFlyAvailable) {
1078 _readyToFlyAvailable = true;
1079 emit readyToFlyAvailableChanged(true);
1080 }
1081
1082 bool newReadyToFly = sysStatus.onboard_control_sensors_health & MAV_SYS_STATUS_PREARM_CHECK;
1083 if (newReadyToFly != _readyToFly) {
1084 _readyToFly = newReadyToFly;
1085 emit readyToFlyChanged(_readyToFly);
1086 }
1087 }
1088
1089 bool newAllSensorsHealthy = (sysStatus.onboard_control_sensors_enabled & sysStatus.onboard_control_sensors_health) == sysStatus.onboard_control_sensors_enabled;
1090 if (newAllSensorsHealthy != _allSensorsHealthy) {
1091 _allSensorsHealthy = newAllSensorsHealthy;
1092 emit allSensorsHealthyChanged(_allSensorsHealthy);
1093 }
1094
1095 if (_onboardControlSensorsPresent != sysStatus.onboard_control_sensors_present) {
1096 _onboardControlSensorsPresent = sysStatus.onboard_control_sensors_present;
1097 emit sensorsPresentBitsChanged(_onboardControlSensorsPresent);
1098 emit requiresGpsFixChanged();
1099 }
1100 if (_onboardControlSensorsEnabled != sysStatus.onboard_control_sensors_enabled) {
1101 _onboardControlSensorsEnabled = sysStatus.onboard_control_sensors_enabled;
1102 emit sensorsEnabledBitsChanged(_onboardControlSensorsEnabled);
1103 }
1104 if (_onboardControlSensorsHealth != sysStatus.onboard_control_sensors_health) {
1105 _onboardControlSensorsHealth = sysStatus.onboard_control_sensors_health;
1106 emit sensorsHealthBitsChanged(_onboardControlSensorsHealth);
1107 }
1108
1109 // ArduPilot firmare has a strange case when ARMING_REQUIRE=0. This means the vehicle is always armed but the motors are not
1110 // really powered up until the safety button is pressed. Because of this we can't depend on the heartbeat to tell us the true
1111 // armed (and dangerous) state. We must instead rely on SYS_STATUS telling us that the motors are enabled.
1112 if (apmFirmware() && _apmArmingNotRequired()) {
1113 _updateArmed(_onboardControlSensorsEnabled & MAV_SYS_STATUS_SENSOR_MOTOR_OUTPUTS);
1114 }
1115
1116 uint32_t newSensorsUnhealthy = _onboardControlSensorsEnabled & ~_onboardControlSensorsHealth;
1117 if (newSensorsUnhealthy != _onboardControlSensorsUnhealthy) {
1118 _onboardControlSensorsUnhealthy = newSensorsUnhealthy;
1119 emit sensorsUnhealthyBitsChanged(_onboardControlSensorsUnhealthy);
1120 }

1127
1128 if (!_lowestBatteryChargeStateAnnouncedMap.contains(batteryStatus.id)) {
1129 _lowestBatteryChargeStateAnnouncedMap[batteryStatus.id] = batteryStatus.charge_state;
1130 }
1131
1132 QString batteryMessage;
1133
1134 switch (batteryStatus.charge_state) {
1135 case MAV_BATTERY_CHARGE_STATE_OK:
1136 _lowestBatteryChargeStateAnnouncedMap[batteryStatus.id] = batteryStatus.charge_state;
1137 break;
1138 case MAV_BATTERY_CHARGE_STATE_LOW:
1139 if (batteryStatus.charge_state > _lowestBatteryChargeStateAnnouncedMap[batteryStatus.id]) {
1140 _lowestBatteryChargeStateAnnouncedMap[batteryStatus.id] = batteryStatus.charge_state;
1141 batteryMessage = tr("battery %1 level low");
1142 }
1143 break;
1144 case MAV_BATTERY_CHARGE_STATE_CRITICAL:
1145 if (batteryStatus.charge_state > _lowestBatteryChargeStateAnnouncedMap[batteryStatus.id]) {
1146 _lowestBatteryChargeStateAnnouncedMap[batteryStatus.id] = batteryStatus.charge_state;
1147 batteryMessage = tr("battery %1 level is critical");
1148 }
1149 break;
1150 case MAV_BATTERY_CHARGE_STATE_EMERGENCY:
1151 if (batteryStatus.charge_state > _lowestBatteryChargeStateAnnouncedMap[batteryStatus.id]) {
1152 _lowestBatteryChargeStateAnnouncedMap[batteryStatus.id] = batteryStatus.charge_state;
1153 batteryMessage = tr("battery %1 level emergency");
1154 }
1155 break;
1156 case MAV_BATTERY_CHARGE_STATE_FAILED:
1157 if (batteryStatus.charge_state > _lowestBatteryChargeStateAnnouncedMap[batteryStatus.id]) {
1158 _lowestBatteryChargeStateAnnouncedMap[batteryStatus.id] = batteryStatus.charge_state;
1159 batteryMessage = tr("battery %1 failed");
1160 }
1161 break;
1162 case MAV_BATTERY_CHARGE_STATE_UNHEALTHY:
1163 if (batteryStatus.charge_state > _lowestBatteryChargeStateAnnouncedMap[batteryStatus.id]) {
1164 _lowestBatteryChargeStateAnnouncedMap[batteryStatus.id] = batteryStatus.charge_state;
1165 batteryMessage = tr("battery %1 unhealthy");
1166 }
1167 break;
1168 }
1169
1170 if (!batteryMessage.isEmpty()) {
1171 QString batteryIdStr("%1");
1172 if (_batteryFactGroupListModel->count() > 1) {
1173 batteryIdStr = batteryIdStr.arg(batteryStatus.id);
1174 } else {
1175 batteryIdStr = batteryIdStr.arg("");
1176 }
1177 _say(tr("warning"));
1178 _say(QStringLiteral("%1 %2 ").arg(_vehicleIdSpeech()).arg(batteryMessage.arg(batteryIdStr)));
1179 }

1268void Vehicle::_handleHeartbeat(mavlink_message_t& message)
1269{
1270 if (message.compid != _defaultComponentId) {
1271 return;
1272 }
1273
1274 mavlink_heartbeat_t heartbeat;
1275
1276 mavlink_msg_heartbeat_decode(&message, &heartbeat);
1277
1278 bool newArmed = heartbeat.base_mode & MAV_MODE_FLAG_DECODE_POSITION_SAFETY;
1279
1280 // ArduPilot firmare has a strange case when ARMING_REQUIRE=0. This means the vehicle is always armed but the motors are not
1281 // really powered up until the safety button is pressed. Because of this we can't depend on the heartbeat to tell us the true
1282 // armed (and dangerous) state. We must instead rely on SYS_STATUS telling us that the motors are enabled.
1283 if (apmFirmware()) {
1284 if (!_apmArmingNotRequired() || !(_onboardControlSensorsPresent & MAV_SYS_STATUS_SENSOR_MOTOR_OUTPUTS)) {
1285 // If ARMING_REQUIRE!=0 or we haven't seen motor output status yet we use the hearbeat info for armed
1286 _updateArmed(newArmed);
1287 }
1288 } else {
1289 // Non-ArduPilot always updates from armed state in heartbeat
1290 _updateArmed(newArmed);
1291 }
1292
1293 if (heartbeat.base_mode != _base_mode || heartbeat.custom_mode != _custom_mode) {
1294 QString previousFlightMode;
1295 if (_base_mode != 0 || _custom_mode != 0){
1296 // Vehicle is initialized with _base_mode=0 and _custom_mode=0. Don't pass this to flightMode() since it will complain about
1297 // bad modes while unit testing.
1298 previousFlightMode = flightMode();
1299 }
1300 _base_mode = heartbeat.base_mode;
1301 _custom_mode = heartbeat.custom_mode;
1302 if (previousFlightMode != flightMode()) {
1303 emit flightModeChanged(flightMode());
1304 }
1305 }

1327void Vehicle::_handleRCChannels(mavlink_message_t& message)
1328{
1329 mavlink_rc_channels_t channels;
1330
1331 mavlink_msg_rc_channels_decode(&message, &channels);
1332
1333 QVector<uint16_t> rawChannelValues({
1334 channels.chan1_raw,
1335 channels.chan2_raw,
1336 channels.chan3_raw,
1337 channels.chan4_raw,
1338 channels.chan5_raw,
1339 channels.chan6_raw,
1340 channels.chan7_raw,
1341 channels.chan8_raw,
1342 channels.chan9_raw,
1343 channels.chan10_raw,
1344 channels.chan11_raw,
1345 channels.chan12_raw,
1346 channels.chan13_raw,
1347 channels.chan14_raw,
1348 channels.chan15_raw,
1349 channels.chan16_raw,
1350 channels.chan17_raw,
1351 channels.chan18_raw,
1352 });
1353
1354 // The internals of radio calibration can ony deal with contiguous channels (other stuff as well!)
1355 int validChannelCount = 0;
1356 int firstUnusedChannelIndex = -1;
1357 for (int i=0; i<rawChannelValues.size(); i++) {
1358 if (rawChannelValues[i] != UINT16_MAX) {
1359 validChannelCount++;
1360 } else if (firstUnusedChannelIndex == -1) {
1361 firstUnusedChannelIndex = i;
1362 }
1363 }
1364 if (firstUnusedChannelIndex != -1 && firstUnusedChannelIndex != validChannelCount) {
1365 qCWarning(VehicleLog) << "Non-contiguous RC channels detected. Not publishing data from RC_CHANNELS.";
1366 return;
1367 }
1368
1369 QVector<int> channelValues(validChannelCount);
1370 QVector<int> clampedValues(validChannelCount);
1371 for (int channelIndex = 0; channelIndex < validChannelCount; ++channelIndex) {
1372 channelValues[channelIndex] = rawChannelValues[channelIndex];
1373 clampedValues[channelIndex] = std::clamp(channelValues[channelIndex], 1000, 2000);
1374 }
1375
1376 // rcRSSI is now a Fact on VehicleFactGroup (this); VehicleFactGroup owns the low-pass
1377 // filter and sentinel handling for the 0-100 / 255-unknown semantics.
1378 updateRCRSSI(channels.rssi);
1379 emit rcChannelsRawChanged(channelValues);
1380 emit rcChannelsClampedChanged(clampedValues);
1381}
