150QString APMAutoPilotPlugin::prerequisiteSetup(VehicleComponent *component) const
151{
152 bool requiresAirframeCheck = false;
153
154 if (qobject_cast<const APMFlightModesComponent*>(component)) {
155 if (_airframeComponent && !_airframeComponent->setupComplete()) {
156 return _airframeComponent->name();
157 }
158 if (_radioComponent && !_radioComponent->setupComplete()) {
159 return _radioComponent->name();
160 }
161 requiresAirframeCheck = true;
162 } else if (qobject_cast<const APMRadioComponent*>(component)) {
163 requiresAirframeCheck = true;
164 } else if (qobject_cast<const APMPowerComponent*>(component)) {
165 requiresAirframeCheck = true;
166 } else if (qobject_cast<const APMSafetyComponent*>(component)) {
167 requiresAirframeCheck = true;
168 } else if (qobject_cast<const APMTuningComponent*>(component)) {
169 requiresAirframeCheck = true;
170 } else if (qobject_cast<const APMSensorsComponent*>(component)) {
171 requiresAirframeCheck = true;
172 }
173
174 if (requiresAirframeCheck) {
175 if (_airframeComponent && !_airframeComponent->setupComplete()) {
176 return _airframeComponent->name();
177 }
178 }
179
180 return QString();

184void APMAutoPilotPlugin::_checkForBadCubeBlack(bool parametersReady)
185{
186 if (!parametersReady) {
187 return;
188 }
189
190 const SharedLinkInterfacePtr sharedLink = _vehicle->vehicleLinkManager()->primaryLink().lock();
191 if (!sharedLink) {
192 return;
193 }
194
195 if (sharedLink->linkConfiguration()->type() != LinkConfiguration::TypeSerial) {
196 return;
197 }
198
199 const SerialLink *serialLink = qobject_cast<const SerialLink*>(sharedLink.get());
200 if (!serialLink) {
201 return;
202 }
203
204 if (!QGCSerialPortInfo(*serialLink->port()).isBlackCube()) {
205 return;
206 }
207
208 ParameterManager *const paramMgr = _vehicle->parameterManager();
209
210 static const QString paramAcc3 = QStringLiteral("INS_ACC3_ID");
211 static const QString paramGyr3 = QStringLiteral("INS_GYR3_ID");
212 static const QString paramEnableMask = QStringLiteral("INS_ENABLE_MASK");
213
214 if (paramMgr->parameterExists(-1, paramAcc3) && (paramMgr->getParameter(-1, paramAcc3)->rawValue().toInt() == 0) &&
215 paramMgr->parameterExists(-1, paramGyr3) && (paramMgr->getParameter(-1, paramGyr3)->rawValue().toInt() == 0) &&
216 paramMgr->parameterExists(-1, paramEnableMask) && (paramMgr->getParameter(-1, paramEnableMask)->rawValue().toInt() >= 7)) {
217 qgcApp()->showAppMessage(tr(
218 "WARNING: The flight board you are using has a critical service bulletin against it which advises against flying. "
219 "For details see: https://discuss.cubepilot.org/t/sb-0000002-critical-service-bulletin-for-cubes-purchased-between-january-2019-to-present-do-not-fly/406"
220 ));
221 }
222}
