233 (controller.channelCount < controller.minChannelCount ?
234 qsTr("%1 channels or more are needed to fly.").arg(controller.minChannelCount) :
235 qsTr("Ready to calibrate.")))
236 return
237 } else if (!controller.joystickMode) {
238 QGroundControl.showMessageDialog(root, qsTr("Zero Trims"),
239 qsTr("Before calibrating you should zero all your trims and subtrims. Click Ok to start Calibration.\n\n%1").arg(
240 (QGroundControl.multiVehicleManager.activeVehicle.px4Firmware ? "" : qsTr("Please ensure all motor power is disconnected AND all props are removed from the vehicle."))),
241 Dialog.Ok,
