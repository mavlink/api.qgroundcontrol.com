146void JoystickConfigController::_readStoredCalibrationValues()
147{
148 if (!_joystick) {
149 qCWarning(JoystickConfigControllerLog) << "Internal Error: No joystick to read calibration for";
150 return;
151 }
152
153 setTransmitterMode(_joystick->settings()->transmitterMode()->rawValue().toInt());
154
155 // Initialize all function mappings to not set
156 for (int i = 0; i < _chanMax; i++) {
157 ChannelInfo *const info = &_rgChannelInfo[i];
158 info->stickFunction = stickFunctionMax;
159 }
160 for (size_t i = 0; i < stickFunctionMax; i++) {
161 _rgFunctionChannelMapping[i] = _chanMax;
162 }
163
164 for (int i = 0; i < _joystick->axisCount(); ++i) {
165 ChannelInfo *const info = &_rgChannelInfo[i];
166 Joystick::AxisCalibration_t calibration = _joystick->getAxisCalibration(i);
167
168 info->channelTrim = calibration.center;
169 info->channelMin = calibration.min;
170 info->channelMax = calibration.max;
171 info->channelReversed = calibration.reversed;
172 info->deadband = calibration.deadband;
173 }
174
175 for (int stickFunctionInt = 0; stickFunctionInt < Joystick::maxAxisFunction; stickFunctionInt++) {
176 auto stickFunction = static_cast<StickFunction>(stickFunctionInt);
177 int functionChannel = _joystick->getChannelForFunction(stickFunction);
178 if (functionChannel >= 0 && functionChannel < _chanMax) {
179 _rgFunctionChannelMapping[stickFunction] = functionChannel;
180 _rgChannelInfo[functionChannel].stickFunction = stickFunction;
181 }
182 }
183
184 _signalAllAttitudeValueChanges();
185}
