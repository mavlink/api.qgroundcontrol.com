150bool PX4FirmwarePlugin::isCapable(const Vehicle *vehicle, FirmwareCapabilities capabilities) const
151{
152 int available = SetFlightModeCapability | PauseVehicleCapability | GuidedModeCapability;
153 //-- This is arbitrary until I find how to really tell if ROI is avaiable
154 if (vehicle->multiRotor()) {
155 available |= ROIModeCapability | ChangeHeadingCapability;
156 }
157 if (vehicle->multiRotor() || vehicle->vtol()) {
158 available |= TakeoffVehicleCapability | GuidedTakeoffCapability | OrbitModeCapability;
159 }
160 if (vehicle->fixedWing()) {
161 available |= TakeoffVehicleCapability | OrbitModeCapability;
162 }
163 return (capabilities & available) == capabilities;
164}

302void PX4FirmwarePlugin::_mavCommandResult(int vehicleId, int component, int command, int result, int failureCode)
303{
304 Q_UNUSED(vehicleId); Q_UNUSED(component); Q_UNUSED(failureCode);
305
306 auto* vehicle = qobject_cast<Vehicle*>(sender());
307 if (!vehicle) {
308 qWarning() << "Dynamic cast failed!";
309 return;
310 }
311
312 if (command == MAV_CMD_NAV_TAKEOFF && result == MAV_RESULT_ACCEPTED) {
313 // Now that we are in takeoff mode we can arm the vehicle which will cause it to takeoff.
314 // We specifically don't retry arming if it fails. This way we don't fight with the user if
315 // They are trying to disarm.
316 disconnect(vehicle, &Vehicle::mavCommandResult, this, &PX4FirmwarePlugin::_mavCommandResult);
317 if (!vehicle->armed()) {
318 vehicle->setArmedShowError(true);
319 }
320 }
321}

425typedef struct {
426 PX4FirmwarePlugin* plugin;
427 Vehicle* vehicle;
428 double newAMSLAlt;
429} PauseVehicleThenChangeAltData_t;
430
431static void _pauseVehicleThenChangeAltResultHandler(void* resultHandlerData, int /*compId*/, const mavlink_command_ack_t& ack, Vehicle::MavCmdResultFailureCode_t failureCode)
432{
433 if (ack.result != MAV_RESULT_ACCEPTED) {
434 switch (failureCode) {
435 case Vehicle::MavCmdResultCommandResultOnly:
436 qDebug() << QStringLiteral("MAV_CMD_DO_REPOSITION error(%1)").arg(ack.result);
437 break;
438 case Vehicle::MavCmdResultFailureNoResponseToCommand:
439 qDebug() << "MAV_CMD_DO_REPOSITION no response from vehicle";
440 break;
441 case Vehicle::MavCmdResultFailureDuplicateCommand:
442 qDebug() << "Internal Error: MAV_CMD_DO_REPOSITION could not be sent due to duplicate command";
443 break;
444 }
445 }
446
447 PauseVehicleThenChangeAltData_t* pData = static_cast<PauseVehicleThenChangeAltData_t*>(resultHandlerData);
448 pData->plugin->_changeAltAfterPause(resultHandlerData, ack.result == MAV_RESULT_ACCEPTED /* pauseSucceeded */);
449}

451void PX4FirmwarePlugin::_changeAltAfterPause(void* resultHandlerData, bool pauseSucceeded)
452{
453 PauseVehicleThenChangeAltData_t* pData = static_cast<PauseVehicleThenChangeAltData_t*>(resultHandlerData);
454
455 if (pauseSucceeded) {
456 pData->vehicle->sendMavCommand(
457 pData->vehicle->defaultComponentId(),
458 MAV_CMD_DO_REPOSITION,
459 true, // show error is fails
460 -1.0f, // Don't change groundspeed
461 MAV_DO_REPOSITION_FLAGS_CHANGE_MODE,
462 0.0f, // Reserved
463 qQNaN(), qQNaN(), qQNaN(), // No change to yaw, lat, lon
464 static_cast<float>(pData->newAMSLAlt));
465 } else {
466 qgcApp()->showAppMessage(tr("Unable to pause vehicle."));
467 }
468
469 delete pData;
470}

472void PX4FirmwarePlugin::guidedModeChangeAltitude(Vehicle* vehicle, double altitudeChange, bool pauseVehicle)
473{
474 if (!vehicle->homePosition().isValid()) {
475 qgcApp()->showAppMessage(tr("Unable to change altitude, home position unknown."));
476 return;
477 }
478 if (qIsNaN(vehicle->homePosition().altitude())) {
479 qgcApp()->showAppMessage(tr("Unable to change altitude, home position altitude unknown."));
480 return;
481 }
482
483 double currentAltRel = vehicle->altitudeRelative()->rawValue().toDouble();
484 double newAltRel = currentAltRel + altitudeChange;
485
486 PauseVehicleThenChangeAltData_t* resultData = new PauseVehicleThenChangeAltData_t;
487 resultData->plugin = this;
488 resultData->vehicle = vehicle;
489 resultData->newAMSLAlt = vehicle->homePosition().altitude() + newAltRel;
490
491 if (pauseVehicle) {
492 Vehicle::MavCmdAckHandlerInfo_t handlerInfo = {};
493 handlerInfo.resultHandler = _pauseVehicleThenChangeAltResultHandler;
494 handlerInfo.resultHandlerData = resultData;
495
496 vehicle->sendMavCommandWithHandler(
497 &handlerInfo,
498 vehicle->defaultComponentId(),
499 MAV_CMD_DO_REPOSITION,
500 -1.0f, // Don't change groundspeed
501 MAV_DO_REPOSITION_FLAGS_CHANGE_MODE,
502 0.0f, // Reserved
503 qQNaN(), qQNaN(), qQNaN(), qQNaN()); // No change to yaw, lat, lon, alt
504 } else {
505 _changeAltAfterPause(resultData, true /* pauseSucceeded */);
506 }
507}

658void PX4FirmwarePlugin::_handleAutopilotVersion(Vehicle* vehicle, mavlink_message_t* message)
659{
660 Q_UNUSED(vehicle);
661
662 auto* instanceData = qobject_cast<PX4FirmwarePluginInstanceData*>(vehicle->firmwarePluginInstanceData());
663 if (!instanceData->versionNotified) {
664 bool notifyUser = false;
665 int supportedMajorVersion = 1;
666 int supportedMinorVersion = 4;
667 int supportedPatchVersion = 1;
668
669 mavlink_autopilot_version_t version;
670 mavlink_msg_autopilot_version_decode(message, &version);
671
672 if (version.flight_sw_version != 0) {
673 int majorVersion, minorVersion, patchVersion;
674
675 majorVersion = (version.flight_sw_version >> (8*3)) & 0xFF;
676 minorVersion = (version.flight_sw_version >> (8*2)) & 0xFF;
677 patchVersion = (version.flight_sw_version >> (8*1)) & 0xFF;
678
679 if (majorVersion < supportedMajorVersion) {
680 notifyUser = true;
681 } else if (majorVersion == supportedMajorVersion) {
682 if (minorVersion < supportedMinorVersion) {
683 notifyUser = true;
684 } else if (minorVersion == supportedMinorVersion) {
685 notifyUser = patchVersion < supportedPatchVersion;
686 }
687 }
688 } else {
689 notifyUser = true;
690 }
691
692 if (notifyUser) {
693 instanceData->versionNotified = true;
694 qgcApp()->showAppMessage(tr("QGroundControl supports PX4 Pro firmware Version %1.%2.%3 and above. You are using a version prior to that which will lead to unpredictable results. Please upgrade your firmware.").arg(supportedMajorVersion).arg(supportedMinorVersion).arg(supportedPatchVersion));
695 }
696 }
697}

724QString PX4FirmwarePlugin::getHobbsMeter(Vehicle* vehicle) const
725{
726 static const char* HOOBS_HI = "LND_FLIGHT_T_HI";
727 static const char* HOOBS_LO = "LND_FLIGHT_T_LO";
728 uint64_t hobbsTimeSeconds = 0;
729
730 if (vehicle->parameterManager()->parameterExists(ParameterManager::defaultComponentId, HOOBS_HI) &&
731 vehicle->parameterManager()->parameterExists(ParameterManager::defaultComponentId, HOOBS_LO)) {
732 Fact* factHi = vehicle->parameterManager()->getParameter(ParameterManager::defaultComponentId, HOOBS_HI);
733 Fact* factLo = vehicle->parameterManager()->getParameter(ParameterManager::defaultComponentId, HOOBS_LO);
734 hobbsTimeSeconds = ((uint64_t)factHi->rawValue().toUInt() << 32 | (uint64_t)factLo->rawValue().toUInt()) / 1000000;
735 qCDebug(VehicleLog) << "Hobbs Meter raw PX4:" << "(" << factHi->rawValue().toUInt() << factLo->rawValue().toUInt() << ")";
736 }
737
738 int hours = hobbsTimeSeconds / 3600;
739 int minutes = (hobbsTimeSeconds % 3600) / 60;
740 int seconds = hobbsTimeSeconds % 60;
741 QString timeStr = QString::asprintf("%04d:%02d:%02d", hours, minutes, seconds);
742 qCDebug(VehicleLog) << "Hobbs Meter string:" << timeStr;
743 return timeStr;
744}
