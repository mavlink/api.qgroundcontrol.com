218void GstVideoReceiver::stop()
219{
220 if (_needDispatch()) {
221 _worker->dispatch([this]() { stop(); });
222 return;
223 }
224
225 if (_uri.isEmpty()) {
226 qCWarning(GstVideoReceiverLog) << "Stop called on empty URI";
227 return;
228 }
229
230 qCDebug(GstVideoReceiverLog) << "Stopping" << _uri;
231
232 if (_teeProbeId != 0) {
233 if (_tee) {
234 GstPad *sinkpad = gst_element_get_static_pad(_tee, "sink");
235 if (sinkpad) {
236 gst_pad_remove_probe(sinkpad, _teeProbeId);
237 gst_clear_object(&sinkpad);
238 }
239 }
240 _teeProbeId = 0;
241 }
242
243 if (_pipeline) {
244 GstBus *bus = gst_pipeline_get_bus(GST_PIPELINE(_pipeline));
245 if (bus) {
246 gst_bus_disable_sync_message_emission(bus);
247 (void) g_signal_handlers_disconnect_by_data(bus, this);
248
249 gboolean recordingValveClosed = TRUE;
250 g_object_get(_recorderValve, "drop", &recordingValveClosed, nullptr);
251
252 if (!recordingValveClosed) {
253 (void) gst_element_send_event(_pipeline, gst_event_new_eos());
254
255 GstMessage *msg = gst_bus_timed_pop_filtered(bus, GST_CLOCK_TIME_NONE, (GstMessageType)(GST_MESSAGE_EOS | GST_MESSAGE_ERROR));
256 if (msg) {
257 switch (GST_MESSAGE_TYPE(msg)) {
258 case GST_MESSAGE_EOS:
259 qCDebug(GstVideoReceiverLog) << "End of stream received!";
260 break;
261 case GST_MESSAGE_ERROR:
262 qCCritical(GstVideoReceiverLog) << "Error stopping pipeline!";
263 break;
264 default:
265 break;
266 }
267
268 gst_clear_message(&msg);
269 } else {
270 qCCritical(GstVideoReceiverLog) << "gst_bus_timed_pop_filtered() failed";
271 }
272 }
273
274 gst_clear_object(&bus);
275 } else {
276 qCCritical(GstVideoReceiverLog) << "gst_pipeline_get_bus() failed";
277 }
278
279 (void) gst_element_set_state(_pipeline, GST_STATE_NULL);
280
281 // FIXME: check if branch is connected and remove all elements from branch
282 if (_fileSink) {
283 _shutdownRecordingBranch();
284 }
285
286 if (_videoSink) {
287 _shutdownDecodingBranch();
288 }
289
290 GST_DEBUG_BIN_TO_DOT_FILE(GST_BIN(_pipeline), GST_DEBUG_GRAPH_SHOW_ALL, "pipeline-stopped");
291
292 gst_clear_object(&_pipeline);
293 _pipeline = nullptr;
294
295 _recorderValve = nullptr;
296 _decoderValve = nullptr;
297 _tee = nullptr;
298 _source = nullptr;
299
300 _lastSourceFrameTime = 0;
301
302 if (_streaming) {
303 _streaming = false;
304 qCDebug(GstVideoReceiverLog) << "Streaming stopped" << _uri;
305 _dispatchSignal([this]() { emit streamingChanged(_streaming); });
306 } else {
307 qCDebug(GstVideoReceiverLog) << "Streaming did not start" << _uri;
308 }
309 }
310
311 qCDebug(GstVideoReceiverLog) << "Stopped" << _uri;
312
313 _dispatchSignal([this]() { emit onStopComplete(STATUS_OK); });

416void GstVideoReceiver::startRecording(const QString &videoFile, FILE_FORMAT format)
417{
418 if (_needDispatch()) {
419 const QString cachedVideoFile = videoFile;
420 _worker->dispatch([this, cachedVideoFile, format]() { startRecording(cachedVideoFile, format); });

424 qCDebug(GstVideoReceiverLog) << "Starting recording" << _uri;

427 qCDebug(GstVideoReceiverLog) << "Streaming is not active!" << _uri;
428 _dispatchSignal([this](){ emit onStartRecordingComplete(STATUS_INVALID_STATE); });
429 return;
430 }

432 if (_recording) {
433 qCDebug(GstVideoReceiverLog) << "Already recording!" << _uri;
434 _dispatchSignal([this]() { emit onStartRecordingComplete(STATUS_INVALID_STATE); });
435 return;
436 }
437
438 qCDebug(GstVideoReceiverLog) << "New video file:" << videoFile << _uri;
439
440 _fileSink = _makeFileSink(videoFile, format);
441 if (!_fileSink) {
442 qCCritical(GstVideoReceiverLog) << "_makeFileSink() failed" << _uri;
443 _dispatchSignal([this]() { emit onStartRecordingComplete(STATUS_FAIL); });
444 return;
445 }
446
447 _removingRecorder = false;
448
449 (void) gst_object_ref(_fileSink);
450
451 gst_bin_add(GST_BIN(_pipeline), _fileSink);
452
453 if (!gst_element_link(_recorderValve, _fileSink)) {
454 qCCritical(GstVideoReceiverLog) << "Failed to link valve and file sink" << _uri;
455 _dispatchSignal([this]() { emit onStartRecordingComplete(STATUS_FAIL); });

457 }
458
459 (void) gst_element_sync_state_with_parent(_fileSink);
460
461 GST_DEBUG_BIN_TO_DOT_FILE(GST_BIN(_pipeline), GST_DEBUG_GRAPH_SHOW_ALL, "pipeline-with-filesink");
462
463 // Install a probe on the recording branch to drop buffers until we hit our first keyframe
464 // When we hit our first keyframe, we can offset the timestamps appropriately according to the first keyframe time
465 // This will ensure the first frame is a keyframe at t=0, and decoding can begin immediately on playback
466 GstPad *probepad = gst_element_get_static_pad(_recorderValve, "src");

471 }
472
473 (void) gst_pad_add_probe(probepad, GST_PAD_PROBE_TYPE_BUFFER, _keyframeWatch, this, nullptr); // to drop the buffers until key frame is received
474 gst_clear_object(&probepad);
475
476 g_object_set(_recorderValve,
477 "drop", FALSE,
478 nullptr);
479
480 _recordingOutput = videoFile;
481 _recording = true;
482 qCDebug(GstVideoReceiverLog) << "Recording started" << _uri;
483 _dispatchSignal([this]() {
484 emit onStartRecordingComplete(STATUS_OK);
485 emit recordingChanged(_recording);
486 });

531void GstVideoReceiver::_watchdog()
532{
533 _worker->dispatch([this]() {
534 if (!_pipeline) {
535 return;
536 }
537
538 const qint64 now = QDateTime::currentSecsSinceEpoch();
539 if (_lastSourceFrameTime == 0) {
540 _lastSourceFrameTime = now;
541 }
542
543 qint64 elapsed = now - _lastSourceFrameTime;
544 if (elapsed > _timeout) {
545 qCDebug(GstVideoReceiverLog) << "Stream timeout, no frames for" << elapsed << _uri;
546 _dispatchSignal([this]() { emit timeout(); });
547 stop();
548 }
549
550 if (_decoding && !_removingDecoder) {
551 if (_lastVideoFrameTime == 0) {
552 _lastVideoFrameTime = now;
553 }
554
555 elapsed = now - _lastVideoFrameTime;
556 if (elapsed > (_timeout * 2)) {
557 qCDebug(GstVideoReceiverLog) << "Video decoder timeout, no frames for" << elapsed << _uri;
558 _dispatchSignal([this]() { emit timeout(); });
559 stop();
560 }
561 }
562 });
563}

601 if (gst_structure_has_name(structure, "video/x-h265")) {
602 filter = gst_caps_from_string("video/x-h265");
603 if (gst_caps_can_intersect(srcCaps, filter)) {
604 sinkCaps = gst_caps_from_string("video/x-h265,stream-format=hvc1");
605 }
606 gst_clear_caps(&filter);

610 sinkCaps = gst_caps_from_string("video/x-h264,stream-format=avc");
611 }
612 gst_clear_caps(&filter);
613 }
614
615 if (sinkCaps) {

617 gst_clear_caps(&sinkCaps);

625GstElement *GstVideoReceiver::_makeSource(const QString &input)
626{
627 if (input.isEmpty()) {
628 qCCritical(GstVideoReceiverLog) << "Failed because URI is not specified";
629 return nullptr;
630 }
631
632 const QUrl sourceUrl(input);
633
634 const bool isRtsp = sourceUrl.scheme().startsWith("rtsp", Qt::CaseInsensitive);
635 const bool isUdp264 = input.contains("udp://", Qt::CaseInsensitive);
636 const bool isUdp265 = input.contains("udp265://", Qt::CaseInsensitive);
637 const bool isUdpMPEGTS = input.contains("mpegts://", Qt::CaseInsensitive);
638 const bool isTcpMPEGTS = input.contains("tcp://", Qt::CaseInsensitive);
639
640 GstElement *source = nullptr;
641 GstElement *buffer = nullptr;
642 GstElement *tsdemux = nullptr;
643 GstElement *parser = nullptr;
644 GstElement *bin = nullptr;
645 GstElement *srcbin = nullptr;
646
647 do {
648 if (isRtsp) {
649 if (!GStreamer::isValidRtspUri(input.toUtf8().constData())) {
650 qCCritical(GstVideoReceiverLog) << "Invalid RTSP URI:" << input;
651 break;
652 }
653
654 source = gst_element_factory_make("rtspsrc", "source");
655 if (!source) {
656 qCCritical(GstVideoReceiverLog) << "gst_element_factory_make('rtspsrc') failed";
657 break;
658 }
659
660 g_object_set(source,
661 "location", input.toUtf8().constData(),
662 "latency", 25,
663 nullptr);
664 } else if (isTcpMPEGTS) {
665 source = gst_element_factory_make("tcpclientsrc", "source");
666 if (!source) {
667 qCCritical(GstVideoReceiverLog) << "gst_element_factory_make('tcpclientsrc') failed";
668 break;
669 }
670
671 const QString host = sourceUrl.host();
672 const quint16 port = sourceUrl.port();
673 g_object_set(source,
674 "host", host.toUtf8().constData(),
675 "port", port,
676 nullptr);
677 } else if (isUdp264 || isUdp265 || isUdpMPEGTS) {
678 source = gst_element_factory_make("udpsrc", "source");
679 if (!source) {
680 qCCritical(GstVideoReceiverLog) << "gst_element_factory_make('udpsrc') failed";
681 break;
682 }
683
684 const QString uri = QStringLiteral("udp://%1:%2").arg(sourceUrl.host(), QString::number(sourceUrl.port()));
685 g_object_set(source,
686 "uri", uri.toUtf8().constData(),
687 nullptr);
688
689 GstCaps *caps = nullptr;
690 if (isUdp264) {
691 caps = gst_caps_from_string("application/x-rtp, media=(string)video, clock-rate=(int)90000, encoding-name=(string)H264");
692 if (!caps) {
693 qCCritical(GstVideoReceiverLog) << "gst_caps_from_string() failed";
694 break;
695 }
696 } else if (isUdp265) {
697 caps = gst_caps_from_string("application/x-rtp, media=(string)video, clock-rate=(int)90000, encoding-name=(string)H265");
698 if (!caps) {
699 qCCritical(GstVideoReceiverLog) << "gst_caps_from_string() failed";
700 break;
701 }
702 }
703
704 if (caps) {
705 g_object_set(source,
706 "caps", caps,
707 nullptr);
708 gst_clear_caps(&caps);
709 }
710 } else {
711 qCDebug(GstVideoReceiverLog) << "URI is not recognized";
712 }
713
714 if (!source) {
715 qCCritical(GstVideoReceiverLog) << "gst_element_factory_make() for data source failed";
716 break;
717 }
718
719 bin = gst_bin_new("sourcebin");
720 if (!bin) {
721 qCCritical(GstVideoReceiverLog) << "gst_bin_new('sourcebin') failed";
722 break;
723 }
724
725 parser = gst_element_factory_make("parsebin", "parser");
726 if (!parser) {
727 qCCritical(GstVideoReceiverLog) << "gst_element_factory_make('parsebin') failed";
728 break;
729 }
730
731 // GStreamer < 1.28: decodebin3 didn't negotiate stream-format caps properly
732 // between parser and decoder, so we forced hvc1/avc. GStreamer 1.28+ fixes
733 // this natively, and the forced caps break hardware decoders that need
734 // byte-stream format (e.g. Qualcomm AMC on Android, D3D12 on Windows).
735#if !defined(QGC_GST_BUILD_VERSION_MAJOR) || (QGC_GST_BUILD_VERSION_MAJOR == 1 && QGC_GST_BUILD_VERSION_MINOR < 28)
736 (void) g_signal_connect(parser, "autoplug-query", G_CALLBACK(_filterParserCaps), nullptr);
737#endif
738
739 gst_bin_add_many(GST_BIN(bin), source, parser, nullptr);
740
741 // FIXME: AV: Android does not determine MPEG2-TS via parsebin - have to explicitly state which demux to use
742 // FIXME: AV: tsdemux handling is a bit ugly - let's try to find elegant solution for that later
743 if (isTcpMPEGTS || isUdpMPEGTS) {
744 tsdemux = gst_element_factory_make("tsdemux", nullptr);
745 if (!tsdemux) {
746 qCCritical(GstVideoReceiverLog) << "gst_element_factory_make('tsdemux') failed";
747 break;
748 }
749
750 (void) gst_bin_add(GST_BIN(bin), tsdemux);
751
752 if (!gst_element_link(source, tsdemux)) {
753 qCCritical(GstVideoReceiverLog) << "gst_element_link() failed";
754 break;
755 }
756
757 source = tsdemux;
758 tsdemux = nullptr;
759 }
760
761 int probeRes = 0;
762 (void) gst_element_foreach_src_pad(source, _padProbe, &probeRes);
763
764 if (probeRes & 1) {
765 if ((probeRes & 2) && (_buffer >= 0)) {
766 buffer = gst_element_factory_make("rtpjitterbuffer", nullptr);
767 if (!buffer) {
768 qCCritical(GstVideoReceiverLog) << "gst_element_factory_make('rtpjitterbuffer') failed";
769 break;
770 }
771
772 (void) gst_bin_add(GST_BIN(bin), buffer);
773
774 if (!gst_element_link_many(source, buffer, parser, nullptr)) {
775 qCCritical(GstVideoReceiverLog) << "gst_element_link() failed";
776 break;
777 }
778 } else {
779 if (!gst_element_link(source, parser)) {
780 qCCritical(GstVideoReceiverLog) << "gst_element_link() failed";
781 break;
782 }
783 }
784 } else {
785 (void) g_signal_connect(source, "pad-added", G_CALLBACK(_linkPad), parser);
786 }
787
788 (void) g_signal_connect(parser, "pad-added", G_CALLBACK(_wrapWithGhostPad), nullptr);
789
790 source = tsdemux = buffer = parser = nullptr;
791
792 srcbin = bin;
793 bin = nullptr;
794 } while(0);
795
796 gst_clear_object(&bin);
797 gst_clear_object(&parser);
798 gst_clear_object(&tsdemux);
799 gst_clear_object(&buffer);
800 gst_clear_object(&source);
801
802 return srcbin;
803}

925void GstVideoReceiver::_logDecodebin3SelectedCodec(GstElement *decodebin3)
926{
927 GValue value = G_VALUE_INIT;
928 GstIterator *iter = gst_bin_iterate_elements(GST_BIN(decodebin3));
929 GstElement *child;
930
931 while (gst_iterator_next(iter, &value) == GST_ITERATOR_OK) {
932 child = GST_ELEMENT(g_value_get_object(&value));
933 GstElementFactory *factory = gst_element_get_factory(child);
934
935 if (factory) {
936 gboolean is_decoder = gst_element_factory_list_is_type(factory, GST_ELEMENT_FACTORY_TYPE_DECODER);
937 if (is_decoder) {
938 const gchar *decoderKlass = gst_element_factory_get_klass(factory);
939 GstPluginFeature *feature = GST_PLUGIN_FEATURE(factory);
940 const gchar *featureName = gst_plugin_feature_get_name(feature);
941 const guint rank = gst_plugin_feature_get_rank(feature);
942 bool isHardwareDecoder = GStreamer::isHardwareDecoderFactory(factory);
943
944 QString pluginName = featureName;
945 GstPlugin *plugin = gst_plugin_feature_get_plugin(feature);
946 if (plugin) {
947 pluginName = gst_plugin_get_name(plugin);
948 gst_object_unref(plugin);
949 }
950 qCDebug(GstVideoReceiverLog) << "Decodebin3 selected codec:rank -" << pluginName << "/" << featureName << "-" << decoderKlass << (isHardwareDecoder ? "(HW)" : "(SW)") << ":" << rank;
951
952 // Disable QoS on the internal decoder to prevent cascading
953 // frame drops on live streams. The videodecoder base class
954 // aggressively advances earliest_time after the first late
955 // frame, causing all subsequent frames to be dropped.
956 g_object_set(child, "qos", FALSE, nullptr);
957 qCDebug(GstVideoReceiverLog) << "Disabled QoS on internal decoder" << featureName;
958 }
959 }
960 g_value_reset(&value);
961 }
962 g_value_unset(&value);
963 gst_iterator_free(iter);
964}

1059bool GstVideoReceiver::_addVideoSink(GstPad *pad)
1060{
1061 GstCaps *caps = gst_pad_query_caps(pad, nullptr);
1062
1063 _ensureVideoSinkInPipeline();
1064
1065 GstPad *sinkPad = gst_element_get_static_pad(_videoSink, "sink");
1066 GstPadLinkReturn linkRet = sinkPad ? gst_pad_link(pad, sinkPad) : GST_PAD_LINK_WRONG_HIERARCHY;
1067 if (linkRet != GST_PAD_LINK_OK) {
1068 qCCritical(GstVideoReceiverLog) << "Unable to link decoder pad to video sink, result:" << linkRet;
1069
1070 // Keep retry behavior by fully detaching the sink when link fails.
1071 // _ensureVideoSinkInPipeline() adds it before linking.
1072 GstObject *parent = gst_element_get_parent(_videoSink);
1073 if (parent) {
1074 (void) gst_element_set_state(_videoSink, GST_STATE_NULL);
1075 (void) gst_bin_remove(GST_BIN(_pipeline), _videoSink);
1076 gst_clear_object(&parent);
1077 }
1078
1079 gst_clear_object(&sinkPad);
1080 gst_clear_caps(&caps);
1081 return false;
1082 }
1083 gst_clear_object(&sinkPad);
1084
1085 (void) gst_element_sync_state_with_parent(_videoSink);
1086
1087 // qml6glsink resets max-lateness=0 during state transitions (it renders
1088 // on QML's paint cycle). For live streams with hardware decoders that
1089 // have startup latency, this causes all frames to be silently dropped
1090 // after the first one. Override after state sync so it sticks.
1091 g_object_set(_videoSink, "sync", FALSE, "max-lateness", G_GINT64_CONSTANT(-1), nullptr);
1092
1093 GST_DEBUG_BIN_TO_DOT_FILE(GST_BIN(_pipeline), GST_DEBUG_GRAPH_SHOW_ALL, "pipeline-with-videosink");
1094
1095 // Determine video size. Errors here are non-fatal.
1096 QSize videoSize;
1097 do {
1098 if (!_decoderValve) {
1099 qCCritical(GstVideoReceiverLog) << "Unable to determine video size - _decoderValve is NULL" << _uri;
1100 break;
1101 }
1102
1103 GstPad *valveSrcPad = gst_element_get_static_pad(_decoderValve, "src");
1104 if (!valveSrcPad) {
1105 qCCritical(GstVideoReceiverLog) << "gst_element_get_static_pad() failed";
1106 break;
1107 }
1108
1109 GstCaps *valveSrcPadCaps = gst_pad_query_caps(valveSrcPad, nullptr);
1110 if (!valveSrcPadCaps) {
1111 qCCritical(GstVideoReceiverLog) << "gst_pad_query_caps() failed";
1112 gst_clear_object(&valveSrcPad);
1113 break;
1114 }
1115
1116 const GstStructure *structure = gst_caps_get_structure(valveSrcPadCaps, 0);
1117 if (!structure) {
1118 qCCritical(GstVideoReceiverLog) << "Unable to determine video size - structure is NULL" << _uri;
1119 gst_clear_object(&valveSrcPad);
1120 break;
1121 }
1122
1123 gint width = 0;
1124 gint height = 0;
1125 (void) gst_structure_get_int(structure, "width", &width);
1126 (void) gst_structure_get_int(structure, "height", &height);
1127 videoSize.setWidth(width);
1128 videoSize.setHeight(height);
1129
1130 gst_clear_caps(&valveSrcPadCaps);
1131 gst_clear_object(&valveSrcPad);
1132 } while (false);
1133 _dispatchSignal([this, videoSize]() { emit videoSizeChanged(videoSize); });
1134
1135 gst_clear_caps(&caps);
1136 return true;
1137}

1278gboolean GstVideoReceiver::_onBusMessage(GstBus * /* bus */, GstMessage *msg, gpointer data)
1279{

1281 qCCritical(GstVideoReceiverLog) << "Invalid parameters in _onBusMessage: msg=" << msg << "data=" << data;
1282 return TRUE;
1283 }
1284
1285 GstVideoReceiver *pThis = static_cast<GstVideoReceiver*>(data);
1286
1287 switch (GST_MESSAGE_TYPE(msg)) {
1288 case GST_MESSAGE_ERROR: {
1289 gchar *debug;
1290 GError *error;
1291 gst_message_parse_error(msg, &error, &debug);
1292
1293 if (debug) {
1294 qCDebug(GstVideoReceiverLog) << "GStreamer debug:" << debug;
1295 g_clear_pointer(&debug, g_free);
1296 }
1297
1298 if (error) {

1300 g_clear_error(&error);

1304 qCDebug(GstVideoReceiverLog) << "Stopping because of error";
1305 pThis->stop();
1306 });
1307 break;
1308 }
1309 case GST_MESSAGE_EOS:
1310 pThis->_worker->dispatch([pThis]() {
1311 qCDebug(GstVideoReceiverLog) << "Received EOS";
1312 pThis->_handleEOS();
1313 });
1314 break;
1315 case GST_MESSAGE_ELEMENT: {
1316 const GstStructure *structure = gst_message_get_structure(msg);
1317 if (!gst_structure_has_name(structure, "GstBinForwarded")) {
1318 break;
1319 }
1320
1321 GstMessage *forward_msg = nullptr;
1322 gst_structure_get(structure, "message", GST_TYPE_MESSAGE, &forward_msg, NULL);
1323 if (!forward_msg) {
1324 break;
1325 }
1326
1327 if (GST_MESSAGE_TYPE(forward_msg) == GST_MESSAGE_EOS) {
1328 pThis->_worker->dispatch([pThis]() {
1329 qCDebug(GstVideoReceiverLog) << "Received branch EOS";
1330 pThis->_handleEOS();
1331 });
1332 }
1333
1334 gst_clear_message(&forward_msg);
1335 break;
1336 }
1337 default:
1338 break;
1339 }
1340
1341 return TRUE;
1342}
