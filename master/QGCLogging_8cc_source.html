141void QGCLogging::_flushToDisk()
142{
143 if (_pendingDiskWrites.isEmpty() || _ioError) {
144 return;
145 }
146
147 // Ensure log output enabled and file open
148 if (!_logFile.isOpen()) {
149 if (!qgcApp()->logOutput()) {
150 _pendingDiskWrites.clear();
151 return;
152 }
153
154 const QString saveDirPath = SettingsManager::instance()->appSettings()->crashSavePath();
155 const QDir saveDir(saveDirPath);
156 const QString saveFilePath = saveDir.absoluteFilePath("QGCConsole.log");
157
158 _logFile.setFileName(saveFilePath);
159 if (!_logFile.open(QIODevice::WriteOnly | QIODevice::Append | QIODevice::Text)) {
160 _ioError = true;
161 qgcApp()->showAppMessage(tr("Open console log output file failed %1 : %2").arg(_logFile.fileName(), _logFile.errorString()));
162 return;
163 }
164 }
165
166 // Check size before writing
167 if (_logFile.size() >= kMaxLogFileSize) {
168 _rotateLogs();
169 }
170
171 // Write all pending lines
172 QTextStream out(&_logFile);
173 for (const QString &line : std::as_const(_pendingDiskWrites)) {
174 out << line << '\n';
175 if (out.status() != QTextStream::Ok) {
176 _ioError = true;
177 qCWarning(QGCLoggingLog) << "Error writing to log file:" << _logFile.errorString();
178 break;
179 }
180 }
181 (void) _logFile.flush();
182 _pendingDiskWrites.clear();
183}
