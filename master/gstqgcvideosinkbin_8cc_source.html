154static void
155gst_qgc_video_sink_bin_init(GstQgcVideoSinkBin *self)
156{
157 self->using_d3d11 = FALSE;
158 self->using_appsink = FALSE;
159
160#ifdef QGC_GST_D3D11_SINK
161 // Prefer D3D11 sink on Windows — zero-copy from D3D hardware decoders,
162 // no OpenGL interop needed, renders via Qt's native D3D11 RHI backend.
163 self->d3d11sink = gst_element_factory_make("qml6d3d11sink", NULL);
164 if (self->d3d11sink) {
165 GST_INFO_OBJECT(self, "Using qml6d3d11sink (D3D11 rendering path)");
166
167 if (!gst_bin_add(GST_BIN(self), self->d3d11sink)) {
168 GST_ERROR_OBJECT(self, "Failed to add qml6d3d11sink to bin");
169 gst_object_unref(self->d3d11sink);
170 self->d3d11sink = NULL;
171 } else if (gst_qgc_video_sink_bin_ghost_pad(self, self->d3d11sink)) {
172 self->using_d3d11 = TRUE;
173 return;
174 } else {
175 gst_bin_remove(GST_BIN(self), self->d3d11sink);
176 self->d3d11sink = NULL;
177 }
178 }
179#endif
180
181#if defined(__APPLE__) && defined(__MACH__)
182 // macOS Metal path: use appsink with videoconvert to avoid OpenGL dependency.
183 // Frames are extracted via the new-sample callback and pushed to a QVideoSink,
184 // which renders through Qt's native Metal RHI backend.
185 self->videoconvert = gst_element_factory_make("videoconvert", NULL);
186 self->appsink = gst_element_factory_make("appsink", "qgcappsink");
187 if (self->videoconvert && self->appsink) {
188 // Accept BGRA so QVideoFrame can use a simple single-plane copy
189 GstCaps *caps = gst_caps_from_string("video/x-raw,format=BGRA");
190 g_object_set(self->appsink,
191 "caps", caps,
192 "emit-signals", TRUE,
193 "max-buffers", 2,
194 "drop", TRUE,
195 "sync", FALSE,
196 NULL);
197 gst_caps_unref(caps);
198
199 gst_bin_add_many(GST_BIN(self), self->videoconvert, self->appsink, NULL);
200 if (gst_element_link(self->videoconvert, self->appsink)
201 && gst_qgc_video_sink_bin_ghost_pad(self, self->videoconvert)) {
202 self->using_appsink = TRUE;
203 GST_INFO_OBJECT(self, "Using appsink (macOS Metal rendering path)");
204 return;
205 }
206
207 GST_WARNING_OBJECT(self, "Failed to link appsink path, falling back to GL");
208 gst_bin_remove(GST_BIN(self), self->videoconvert);
209 gst_bin_remove(GST_BIN(self), self->appsink);
210 self->videoconvert = NULL;
211 self->appsink = NULL;
212 } else {
213 GST_WARNING_OBJECT(self, "Failed to create appsink path elements: videoconvert=%p appsink=%p, falling back to GL",
214 (void *)self->videoconvert, (void *)self->appsink);
215 gst_clear_object(&self->videoconvert);
216 gst_clear_object(&self->appsink);
217 }
218#endif
219
220 // GL path: glsinkbin wraps qml6glsink with automatic glupload/glcolorconvert
221 self->glsinkbin = gst_element_factory_make("glsinkbin", NULL);
222 if (!self->glsinkbin) {
223 GST_ERROR_OBJECT(self, "gst_element_factory_make('glsinkbin') failed");
224 return;
225 }
226
227 self->qmlglsink = gst_element_factory_make("qml6glsink", NULL);
228 if (!self->qmlglsink) {
229 GST_ERROR_OBJECT(self, "gst_element_factory_make('qml6glsink') failed");
230 gst_clear_object(&self->glsinkbin);
231 return;
232 }
233
234 // glsinkbin takes ownership of qmlglsink (sinks its floating ref)
235 g_object_set(self->glsinkbin,
236 "sink", self->qmlglsink,
237 PROP_ENABLE_LAST_SAMPLE_NAME, FALSE,
238 NULL);
239
240 if (!gst_bin_add(GST_BIN(self), self->glsinkbin)) {
241 GST_ERROR_OBJECT(self, "Failed to add glsinkbin to bin");
242 // glsinkbin owns qmlglsink — clearing it frees both
243 self->qmlglsink = NULL;
244 gst_clear_object(&self->glsinkbin);
245 return;
246 }
247
248 gst_qgc_video_sink_bin_ghost_pad(self, self->glsinkbin);
249}

262 : self->using_appsink ? self->appsink
263 : self->glsinkbin;
264
265 switch (prop_id) {

268 g_object_set(activeBin,
269 PROP_ENABLE_LAST_SAMPLE_NAME,
270 g_value_get_boolean(value),

272 break;
273 case PROP_WIDGET:
274 if (self->using_appsink)
275 break; // appsink path does not use a widget
276 if (G_LIKELY(activeSink))

278 PROP_WIDGET_NAME,
279 g_value_get_pointer(value),
280 NULL);
281 break;
282 case PROP_FORCE_ASPECT_RATIO:
283 if (self->using_appsink)
284 break;
285 if (G_LIKELY(activeSink))
286 g_object_set(activeSink,
287 PROP_FORCE_ASPECT_RATIO_NAME,
288 g_value_get_boolean(value),
289 NULL);
290 break;
291 case PROP_PIXEL_ASPECT_RATIO: {
292 if (self->using_appsink)
293 break;
294 const gint num = gst_value_get_fraction_numerator(value);
295 const gint den = gst_value_get_fraction_denominator(value);
296 if (G_LIKELY(activeSink))

308 g_value_get_boolean(value),
309 NULL);

311 case PROP_MAX_LATENESS:
312 if (G_LIKELY(activeSink))
313 g_object_set(activeSink,
314 PROP_MAX_LATENESS_NAME,
315 g_value_get_int64(value),
316 NULL);
317 break;

320 break;
